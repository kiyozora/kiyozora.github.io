public class NBody{
	
	public static double readRadius(String path){
		In info = new In(path);
		int num = info.readInt();
		double siRad = info.readDouble();
		return siRad;
	}

	public static Planet[] readPlanets(String path){
		In info = new In(path);
		int num = info.readInt();
		Planet[] list = new Planet[num];
		double siRad = info.readDouble();
		int index = 0;

		while(index < num) {
			double xp = info.readDouble();
			double yp = info.readDouble();
			double xv = info.readDouble();
			double yv = info.readDouble();
			double mass = info.readDouble();
			String name = info.readString();
			list[index] = new Planet(xp, yp, xv, yv, mass, name);
			index ++;
		}
		return list;

	} 

	public static void main(String[] args) {
		double T = Double.parseDouble(args[0]);
		double dt= Double.parseDouble(args[1]);
		String filename = args[2];
		double radius = readRadius(filename);
		int time = 0;
		Planet[] pArray = readPlanets(filename);

		StdDraw.clear();

		StdDraw.picture(0.5,0.5,"./images/starfield.jpg");

		StdDraw.setScale(readRadius(filename)*-1,readRadius(filename));
		for(int i=0;i<pArray.length;i++){
			// StdDraw.picture(planets[i].xxPos,planets[i].yyPos,"./images/"+planets[i].imgFileName);
			pArray[i].draw();
		}

		while (time < T){
			double[] xForces = new double[pArray.length];
			double[] yForces = new double[pArray.length];
			for (int i = 0; i < xForces.length; i++){
				xForces[i] = pArray[i].calcNetForceExertedByX(pArray);
			}
			for (int i = 0; i < yForces.length; i++){
				yForces[i] = pArray[i].calcNetForceExertedByY(pArray);
			}
			for (int i = 0; i < pArray.length; i++){
				pArray[i].update(dt, xForces[i], yForces[i]);
			}

			StdDraw.clear();
			StdDraw.picture(0.5,0.5,"./images/starfield.jpg");
			StdDraw.setScale(readRadius(filename)*-1,readRadius(filename));
			for(int i=0;i<pArray.length;i++){
				pArray[i].draw();
			}

			StdDraw.show(10);	
			time += dt;
		}
		StdOut.printf("%d\n", pArray.length);
		StdOut.printf("%.2e\n", radius);
		for (int i = 0; i < pArray.length; i++) {
		StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
   		pArray[i].xxPos, pArray[i].yyPos, pArray[i].xxVel, pArray[i].yyVel, pArray[i].mass, pArray[i].imgFileName);	
		}		
		
	}
}

public class Planet {
	public double xxPos;
	public double yyPos;
	public double xxVel;
    public double yyVel; 
    public double mass;
    public String imgFileName;
    private static double g=6.67e-11;

    public Planet(double xP, double yP, double xV,
              double yV, double m, String img){
    	xxPos = xP;
    	yyPos = yP;
    	yyVel = yV;
    	xxVel = xV;
    	mass  = m;
    	imgFileName= img;
    }

    public Planet(Planet p){
    	xxPos = p.xxPos;
    	yyPos = p.yyPos;
    	xxVel = p.xxVel;
    	yyVel = p.yyVel;
    	mass = p.mass;
    	imgFileName= p.imgFileName;
    }

    /*
		distance in x,y, and r^2
    */
    private double dx(Planet other_p){
    	return other_p.xxPos - this.xxPos;
    } 

    private double dy(Planet other_p){
    	return other_p.yyPos - this.yyPos;
    }

    private double dist2(Planet other_p){
    	double dist_x = dx(other_p);
    	double dist_y = dy(other_p);
    	return (dist_x*dist_x)+(dist_y*dist_y);
    }

    private double dist(Planet other_p){
    	//System.out.print(dist2(other_p)+ "| ");
    	return Math.sqrt(dist2(other_p));
    }


    /*
    	force / F on X / F on Y
	*/
    private double force(Planet other_p){
    	return g*this.mass*other_p.mass/dist2(other_p);
    }

    private double Fy(Planet other_p){
    	return force(other_p)*dy(other_p)/dist(other_p);
    }

    private double Fx(Planet other_p){
    	return force(other_p)*dx(other_p)/dist(other_p);
    }

    private double Ax(Planet other_p){
    	return Fx(other_p)/mass;
    }

    private double Ay(Planet other_p){
    	return Fy(other_p)/mass;
    }

    /*
    	external methods to access force
    */
    public double calcForceExertedByX(Planet other_p){
    	return Fx(other_p);
    }
    public double calcForceExertedByY(Planet other_p){
    	return Fy(other_p);
    }
    public double calcDistance(Planet other_p){
    	return dist(other_p);
    }
    public double calcForceExertedBy(Planet other_p){
        return force(other_p);
    }

    // deal with arrays of planets
    public double calcNetForceExertedByX(Planet[] others_p){
    	double aggreFx = 0;
    	for(int i = 0;i<others_p.length;i++){
    		if(this.equals(others_p[i]))
    			continue;
    		aggreFx += calcForceExertedByX(others_p[i]);
    	}
    	return aggreFx;
    }
    public double calcNetForceExertedByY(Planet[] others_p){
    	double aggreFy = 0;
    	for(int i = 0; i< others_p.length; i++)
    	{
			if(this.equals(others_p[i]))
    			continue;
    		aggreFy += calcForceExertedByY(others_p[i]);

    	}
    	return aggreFy;
    }

    public void update(double dt,double fX,double fY){
        double aX = fX/this.mass;
        double aY = fY/this.mass;
        xxVel += dt*aX;
        yyVel += dt*aY;
        xxPos += dt*xxVel;
        yyPos += dt*yyVel;

    }


    public void  draw(){
        StdDraw.picture(xxPos,yyPos,"./images/"+imgFileName);
    }
}
