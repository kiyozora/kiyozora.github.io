//Nbody Main Class
public class NBody{

	/** Main method for NBody
	  * Looks through file, sets planets, 
	  * then draws animation
	  */
	public static void main(String[] args) {

		//Setting string of arguments from terminal
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String filename = args[2];

		//Collects num of planets and the radius of the world
		double radius = readRadius(filename);

		//Create an specific number array of planets 
		Planet[] planets = readPlanets(filename);

		//Sets the scale of the universe
		StdDraw.setScale(-radius, radius);


		/**While Loop: (Creating an animation)
		  * - Grab all Net forces
		  * - Update Planets information
		  * - Draw Planet
		  */
		double timer = 0;
		while (timer <= T){

			//Draws the background image
			StdDraw.picture(0, 0, "/images/starfield.jpg");

			//Create an array of xForces and YForces
			double[] xForces = new double[planets.length];
			double[] yForces = new double[planets.length];

			//Collects all xForces and YForces from planets
			for (int i=0; i < planets.length; i+=1){
				xForces[i] = planets[i].calcNetForceExertedByX(planets);
				yForces[i] = planets[i].calcNetForceExertedByY(planets);
			}

			//Updates the position of planets
			for(int i=0; i < planets.length; i+=1){
				planets[i].update(dt, xForces[i], yForces[i]);
			}

			//Draws each planet
			for (Planet p: planets){
				p.draw();
			}

			//Shows frame each 10 seconds, updates time, and prints time left
			StdDraw.show(10);
			timer += dt;
		}

		StdOut.printf("%d\n", planets.length);
		StdOut.printf("%.2e\n", radius);
		for (int i = 0; i < planets.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
		   		planets[i].xxPos, planets[i].yyPos, planets[i].xxVel, planets[i].yyVel, planets[i].mass, planets[i].imgFileName);	
		}	

	}

	public static double readRadius(String planetsTxt) {
		In planetsInfo = new In(planetsTxt);
		planetsInfo.readInt();
		return planetsInfo.readDouble();
	}

	public static Planet[] readPlanets(String planetsTxt) {
		In planetsInfo = new In(planetsTxt);

		int num_planets = planetsInfo.readInt();
		planetsInfo.readDouble();

		Planet[] planets = new Planet[num_planets];
		/** Go through the array and add the necessary information
		 * for each planet from the In file
		 */
		for (int i=0; i < num_planets; i+=1){
			planets[i] = new Planet(planetsInfo.readDouble(), planetsInfo.readDouble(), planetsInfo.readDouble(),
					planetsInfo.readDouble(), planetsInfo.readDouble(), planetsInfo.readString());
		}

		return planets;
	}

}

//Planet Class

public class Planet{

	//Instances of Planet class
	public double xxPos;
	public double yyPos;
	public double xxVel;
	public double yyVel;
	public double mass;
	public String imgFileName;

	//Creating a new Planet object and assigning it's value
	public Planet(double xxPos, double yyPos, double xxVel, 
					double yyVel, double mass, String imgFileName){
		this.xxPos = xxPos;
		this.yyPos = yyPos;
		this.xxVel = xxVel;
		this.yyVel = yyVel;
		this.mass = mass;
		this.imgFileName = imgFileName;
	}


	//Copying a Planet, therefore no reference is being pointed
	public Planet(Planet p){
		this.xxPos = p.xxPos;
		this.yyPos = p.yyPos;
		this.xxVel = p.xxVel;
		this.yyVel = p.yyVel;
		this.mass = p.mass;
		this.imgFileName = p.imgFileName;;
	}

	//Returns the distance between two planets
	public double calcDistance(Planet p2){
		double dx = p2.xxPos - this.xxPos;
		double dy = p2.yyPos - this.yyPos;

		return Math.sqrt((dx * dx) + (dy * dy));
	}

	//Returns force exerted on a planet by another planet
	public double calcForceExertedBy(Planet p2){
		double g = 6.67 * Math.pow(10, -11);
		double r = calcDistance(p2);
				
		return (g * this.mass * p2.mass) / (r * r);
	}

	//Returns force exerted in Y direction from two planets
	public double calcForceExertedByY(Planet p2){
		double f = calcForceExertedBy(p2);
		double dy = p2.yyPos - this.yyPos;
		double r = calcDistance(p2);
		
		return f * dy / r;
	}

	//Returns force exerted in X direction from two planets
	public double calcForceExertedByX(Planet p2){
		double f = calcForceExertedBy(p2);
		double dx = p2.xxPos - this.xxPos;
		double r = calcDistance(p2);
		
		return f * dx / r;
	}

	//Returns the net Y force exerted by an array of Planets
	public double calcNetForceExertedByY(Planet[] planets){
		double total = 0;
		for (Planet p: planets){
			if(this != p){
				total += calcForceExertedByY(p);
			}
		}
		return total;
	}

	//Returns the net X force exerted by an array of Planets
	public double calcNetForceExertedByX(Planet[] planets){
		double total = 0;
		for (Planet p: planets){
			if(this != p){
				total += calcForceExertedByX(p);
			}
		}
		return total;
	}

	//Updates the Position and Velocity of a planet
	public void update(double dt, double fx, double fy){
		double xAccel = fx / this.mass;
		double yAccel = fy / this.mass;
		
		this.xxVel += dt * xAccel;
		this.yyVel += dt * yAccel;
		
		this.xxPos += dt * this.xxVel;
		this.yyPos += dt * this.yyVel;
		
	}

	public void draw(){
		StdDraw.picture(this.xxPos, this.yyPos, "/images/" + imgFileName);
	}
}
