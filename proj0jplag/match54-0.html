public class NBody{
	// Returns radius of universe (second line of .txt file)
	public static double readRadius(String fileName) {
		In in = new In(fileName); // Start reading .txt file
		int planets = in.readInt(); // Read number of planets (first line)
		double radius = in.readDouble(); // Reads radius (second line)
		return radius;
	}

	// Returns an array of planets based on info in .txt file
	public static Planet[] readPlanets(String fileName) {
		In in = new In(fileName); // Start reading .txt file
		int numPlanets = in.readInt(); // Read number of planets (first line)
		double radius = in.readDouble(); // Reads radius (second line)
		Planet[] planets = new Planet[numPlanets];
		for (int i = 0; i < numPlanets; i++) {
			double xP = in.readDouble(); // Read initial x position of planet
			double yP = in.readDouble(); // Read initial y position of planet
			double xV = in.readDouble(); // Read initial x velocity of planet
			double yV = in.readDouble(); // Read initial y velocity of planet
			double mass = in.readDouble(); // Read mass of planet
			String img = in.readString(); // Read name of image file for planet
			planets[i] = new Planet(xP, yP, xV, yV, mass, img);
		}
		return planets;
	}

	public static void main(String[] args) {
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String filename = args[2];
		double radius = readRadius(filename);
		Planet[] planets = readPlanets(filename);
		StdDraw.setScale(-radius, radius);
		StdDraw.picture(0, 0, "images/starfield.jpg");
		for (Planet p : planets) {
			p.draw();
		}
		double time = 0;
		while (time < T) {
			double[] xForces = new double[planets.length];
			double[] yForces = new double[planets.length];
			for (int i = 0; i < planets.length; i++) {
				xForces[i] = planets[i].calcNetForceExertedByX(planets);
				yForces[i] = planets[i].calcNetForceExertedByY(planets);
			}
			for (int i = 0; i < planets.length; i++) {
				planets[i].update(dt, xForces[i], yForces [i]);
			}
			StdDraw.picture(0, 0, "images/starfield.jpg");
			for (Planet p : planets) {
				p.draw();
			}
			StdDraw.show(10);
			time += dt;
		}
		StdOut.printf("%d\n", planets.length);
		StdOut.printf("%.2e\n", radius);
		for (int i = 0; i < planets.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n", 
						  planets[i].xxPos, planets[i].yyPos, planets[i].xxVel, 
						  planets[i].yyVel, planets[i].mass, planets[i].imgFileName);
		}
	}
}

Planet.java

public class Planet{
	public double xxPos; // current x position
	public double yyPos; // current y position
	public double xxVel; // current velocity in x direction
	public double yyVel; // current velocity in y direction
	public double mass; // mass of planet
	public String imgFileName; /* name of an image in the image 
								* directory that depicts the planet */

	// Instantiates Planet class with given instance variables
	public Planet(double xP, double yP, double xV,
				  double yV, double m, String img) {
		this.xxPos = xP;
		this.yyPos = yP;
		this.xxVel = xV;
		this.yyVel = yV;
		this.mass = m;
		this.imgFileName = img;
	}

	// Makes an exact planet copy of given planet
	public Planet(Planet p) {
		this.xxPos = p.xxPos;
		this.yyPos = p.yyPos;
		this.xxVel = p.xxVel;
		this.yyVel = p.yyVel;
		this.mass = p.mass;
		this.imgFileName = p.imgFileName;
	}

	// Calculates the distance between current planet and given planet p
	public double calcDistance(Planet p) {
		double dx = xxPos - p.xxPos;
		double dy = yyPos - p.yyPos;
		double r = Math.sqrt(dx*dx + dy*dy);
		return r;
	}

	// Calculates force exerted by given planet on current planet
	public double calcForceExertedBy(Planet p) {
		double g = 6.67e-11;
		double r = calcDistance(p);
		double force = (g * mass * p.mass) / (r*r);
		return force;
	}

	// Calculates force exerted in x direction by given planet on current
	public double calcForceExertedByX(Planet p) {
		double force = calcForceExertedBy(p);
		double dx = p.xxPos - xxPos;
		double r = calcDistance(p);
		return force * dx / r;
	}

	// Calculates force exerted in y direction by given planet on current
	public double calcForceExertedByY(Planet p) {
		double force = calcForceExertedBy(p);
		double dy = p.yyPos - yyPos;
		double r = calcDistance(p);
		return force * dy / r;
	}

	// Calculates net force exerted in x direction by all planets on current
	public double calcNetForceExertedByX(Planet[] pArray) {
		double netXForce = 0;
		for (Planet p : pArray) {
			if (!this.equals(p)) {
				netXForce += calcForceExertedByX(p);
			}
		}
		return netXForce;
	}

	// Calculates net force exerted in y direction by all planets on current
	public double calcNetForceExertedByY(Planet[] pArray) {
		double netYForce = 0;
		for (Planet p : pArray) {
			if (!this.equals(p)) {
				netYForce += calcForceExertedByY(p);
			}
		}
		return netYForce;
	}

	// Updates position and velocity from forces fX and Fy after a time, dt
	public void update(double dt, double fX, double fY) {
		double ax = fX/mass;
		double ay = fY/mass;
		xxVel += dt*ax;
		yyVel += dt*ay;
		xxPos += dt*xxVel;
		yyPos += dt*yyVel;
	}

	// Draws the planets img at its position
	public void draw() {
		StdDraw.picture(xxPos, yyPos, "images/" + imgFileName);
	}
}
