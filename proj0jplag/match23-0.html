NBody.java

public class NBody {

	public static double readRadius(String text) {
		In in = new In(text);
		in.readDouble();
		return in.readDouble();
	}

	public static Planet[] readPlanets(String text) {
		In in = new In(text);
		int numberPlanets = in.readInt();
		Planet[] arrayPlanets = new Planet[numberPlanets];
		in.readDouble();
		for (int i = 0; i < numberPlanets ; i++) {
			arrayPlanets[i] = new Planet(in.readDouble(), in.readDouble(), in.readDouble(), in.readDouble(), in.readDouble(), in.readString());
		}
		return arrayPlanets;
	}

	public static void main(String[] args) {
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String filename = args[2];
		double radius = readRadius(filename);
		Planet[] planets = readPlanets(filename);
		StdDraw.setScale(-radius, radius);
		StdDraw.picture(0, 0, "images/starfield.jpg");
		for (Planet p : planets) {
			p.draw();
		}
		for (int time = 0; time < T; time++) {
			double[] xForces = new double[planets.length];
			double[] yForces = new double[planets.length];
			for (int n = 0; n < planets.length; n++) {
				xForces[n] = planets[n].calcNetForceExertedByX(planets);
				yForces[n] = planets[n].calcNetForceExertedByY(planets);
			}
			for (int z = 0; z < planets.length; z++) {
				planets[z].update(dt, xForces[z], yForces[z]);
			}
			StdDraw.picture(0, 0, "images/starfield.jpg");
			for (Planet p : planets) {
				p.draw();
			}
			StdDraw.show(10);
			time += dt;
		}	
		StdOut.printf("%d\n", planets.length);
		StdOut.printf("%.2e\n", radius);
		for (int i = 0; i < planets.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
   			planets[i].xxPos, planets[i].yyPos, planets[i].xxVel, planets[i].yyVel, planets[i].mass, planets[i].imgFileName);	
		}
	}
}

Planet.java

public class Planet {

	public double xxPos;
	public double yyPos;
	public double xxVel;
	public double yyVel;
	public double mass;
	public String imgFileName;
	private double gravity = 6.67 * Math.pow(10, -11);

	public Planet(double xP, double yP, double xV, double yV, double m, String img) {
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}

	public Planet(Planet p) {
		this.xxPos = p.xxPos;
		this.yyPos = p.yyPos;
		this.xxVel = p.xxVel;
		this.yyVel = p.yyVel;
		this.mass = p.mass;
		this.imgFileName = p.imgFileName;
	}

	public double calcDistance(Planet z) {
		double dx = this.xxPos - z.xxPos;
		double dy = this.yyPos - z.yyPos;
		return Math.sqrt(dx * dx + dy * dy);
	}

	public double calcForceExertedBy(Planet z) {
		return gravity * this.mass * z.mass / (this.calcDistance(z) * this.calcDistance(z));
	}

	public double calcForceExertedByX(Planet z) {
		return (this.calcForceExertedBy(z) * (z.xxPos - this.xxPos))/ this.calcDistance(z);
	}

	public double calcForceExertedByY(Planet z) {
		return (this.calcForceExertedBy(z) * (z.yyPos - this.yyPos))/ this.calcDistance(z);
	}

	public double calcNetForceExertedByX(Planet[] planets) {
		double netForce = 0;
		for(int i = 0; i < planets.length; i++) {
			if (!this.equals(planets[i])) {
				netForce += this.calcForceExertedByX(planets[i]); 
			}
		}
		return netForce;
	}

	public double calcNetForceExertedByY(Planet[] planets) {
		double netForce = 0;
		for(int i = 0; i < planets.length; i++) {
			if (!this.equals(planets[i])) {
				netForce += this.calcForceExertedByY(planets[i]); 
			}
		}
		return netForce;
	}

	public void update(double dt, double fX, double fY) {
		double aX = fX/this.mass; 
		double aY = fY/this.mass;
		this.xxVel = this.xxVel + dt * aX;
		this.yyVel = this.yyVel + dt * aY;
		this.xxPos = this.xxPos + dt * this.xxVel;
		this.yyPos = this.yyPos + dt * this.yyVel;
	}
	
	public void draw() {
		StdDraw.picture(xxPos, yyPos, "images/" + imgFileName);
	}

}
