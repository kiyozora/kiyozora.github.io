NBody.java

public class NBody
{

    public static void main(String[] args)
    {
        double T = Double.parseDouble(args[0]);
        double dt = Double.parseDouble(args[1]);
        String filename = args[2];
        double radius = readRadius(filename);
        Planet[] planets = readPlanets(filename);
        String bgImg = "./images/starfield.jpg";
        StdDraw.setScale(-radius, radius);

        double time = 0;
        // simulate everything
        while (time < T)
        {
            // calculate net forces on the planets
            double[] xForces = new double[planets.length];
            double[] yForces = new double[planets.length];
            for (int i = 0; i < planets.length; i++)
            {
                xForces[i] = planets[i].calcNetForceExertedByX(planets);
                yForces[i] = planets[i].calcNetForceExertedByY(planets);
            }
            // draw images
            StdDraw.picture(0, 0, bgImg);
            for (int i = 0; i < planets.length; i++)
            {
                planets[i].update(dt, xForces[i], yForces[i]);
                planets[i].draw();
            }
            StdDraw.show(10);
            // increment time
            time += dt;
        }
        // print out the final state of the universe
        StdOut.printf("%d\n", planets.length);
        StdOut.printf("%.2e\n", radius);
        for (int i = 0; i < planets.length; i++) {
        	StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
           		planets[i].xxPos, planets[i].yyPos, planets[i].xxVel,
                planets[i].yyVel, planets[i].mass, planets[i].imgFileName);
        }
    }


    public static double readRadius(String filename)
    {
        In in = new In(filename);
        in.readInt();
        return in.readDouble();
    }

    public static Planet[] readPlanets(String filename)
    {
        In in = new In(filename);
        int planetNumber = in.readInt();    // get the total number of planets
        Planet[] planets = new Planet[planetNumber];
        in.readDouble();
        // create a new planet for every line of info that we read
        for (int i = 0; i < planets.length; i++)
        {
            double xPos = in.readDouble();
            double yPos = in.readDouble();
            double xVel = in.readDouble();
            double yVel = in.readDouble();
            double mass = in.readDouble();
            String img = in.readString();
            planets[i] = new Planet(xPos, yPos, xVel, yVel, mass, img);
        }
        return planets;
    }
}

Planet.java

public class Planet
{
    public double xxPos, yyPos;
    public double xxVel, yyVel;
    public double mass;
    public String imgFileName;


    public Planet(double xP, double yP, double xV, double yV, double m, String img)
    {
        xxPos = xP;
        yyPos = yP;
        xxVel = xV;
        yyVel = yV;
        mass = m;
        imgFileName = img;    // get the right path for the image
    }

    public Planet(Planet p)
    {
        xxPos = p.xxPos;
        yyPos = p.yyPos;
        xxVel = p.xxVel;
        yyVel = p.yyVel;
        mass = p.mass;
        imgFileName = p.imgFileName;
    }

    public double calcDistance(Planet p)
    {
        // use Pythagorean formula
        double xDist = xxPos - p.xxPos;
        double yDist = yyPos - p.yyPos;
        return Math.sqrt(xDist * xDist + yDist * yDist);
    }

    public double calcForceExertedBy(Planet p)
    {
        // use formula for force
        double dist = calcDistance(p);
        return 6.67*Math.pow(10, -11) * mass * p.mass / (dist * dist);

    }

    public double calcForceExertedByX(Planet p)
    {
        double xDist = p.xxPos - xxPos;
        return calcForceExertedBy(p) * xDist / calcDistance(p);
    }

    public double calcForceExertedByY(Planet p)
    {
        double yDist = p.yyPos - yyPos;
        return calcForceExertedBy(p) * yDist / calcDistance(p);
    }

    public double calcNetForceExertedByX(Planet[] ps)
    {
        // calculate x-axis force between two planets, making sure that the planets
        // are different
        double xForce = 0;
        for (Planet p: ps)
            if (!this.equals(p))
                xForce += calcForceExertedByX(p);
        return xForce;
    }

    public double calcNetForceExertedByY(Planet[] ps)
    {
        // calculate y-axis force between two planets, making sure that the planets
        // are different
        double yForce = 0;
        for (Planet p: ps)
            if (!this.equals(p))
                yForce += calcForceExertedByY(p);
        return yForce;
    }

    public void update(double dt, double fX, double fY)
    {
        /** Updates the planet's position and velocity instance variables
        *   dt: duration of applied forces
        *   fX, fY: forces in the X and Y directions, respectively
        */
        double xAccel = fX / mass, yAccel = fY / mass;
        xxVel += dt * xAccel;
        yyVel += dt * yAccel;
        xxPos += dt * xxVel;
        yyPos += dt * yyVel;
    }

    public void draw()
    {
        // draw the planet at its position
        StdDraw.picture(xxPos, yyPos, "./images/" + imgFileName);
    }
}
