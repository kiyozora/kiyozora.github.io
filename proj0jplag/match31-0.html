/** Class that runs a simulation of the universe
 *
 * @author Rachael Wu, Annie Wang
 * @version 01/25/16
 *
 */

public class NBody {
	public static double readRadius(String FileName)
	{
		In in = new In(FileName);
		in.readInt();
		return in.readDouble();
	}

	public static Planet[] readPlanets(String FileName)
	{
		In in = new In(FileName);
		int i = in.readInt();
		in.readDouble();

		Planet[] p = new Planet[i];

		for (int j = 0; j < i; j++)
		{
			p[j] = new Planet(in.readDouble(), in.readDouble(), in.readDouble(), in.readDouble(), 
				in.readDouble(), in.readString());
		}

		return p;
	}

	public static void main(String[] args)
	{
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String filename = args[2];
		double radius = readRadius(filename);
		Planet[] p = readPlanets(filename);

		StdDraw.setScale(-1 * radius, radius);
		StdDraw.picture(0, 0, "images/starfield.jpg");

		for (int i = 0; i < p.length; i++)
		{
			p[i].draw();
		}

		double time = 0;

		while (time < T)
		{
			double[] xForces = new double[p.length];
			double[] yForces = new double[p.length];

			for (int i = 0; i < p.length; i++)
			{
				xForces[i] = p[i].calcNetForceExertedByX(p);
				yForces[i] = p[i].calcNetForceExertedByY(p);
			}

			for (int i = 0; i < p.length; i++)
			{
				p[i].update(dt, xForces[i], yForces[i]);
			}

			StdDraw.picture(0, 0, "images/starfield.jpg");

			for (int i = 0; i < p.length; i++)
			{
				p[i].draw();
			}

			StdDraw.show(10);

			time += dt;
		}

		StdOut.printf("%d\n", p.length);
		StdOut.printf("%.2e\n", radius);

		for (int i = 0; i < p.length; i++) 
		{
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
   			p[i].xxPos, p[i].yyPos, p[i].xxVel, p[i].yyVel, 
   			p[i].mass, p[i].imgFileName);
		}
	}
}

/** Class that simulates a planet
 *
 * @author Rachael Wu, Annie Wang
 * @version 01/25/16
 *
 */

import java.lang.Math;

public class Planet {
	public double xxPos, yyPos, xxVel, yyVel, mass;
	public String imgFileName;
	private static double gravity = 6.67 * Math.pow(10, -11);

	public Planet(double xP, double yP, double xV, double yV, double m, String img)
	{
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}

	public Planet(Planet p)
	{
		xxPos = p.xxPos;
		yyPos = p.yyPos;
		xxVel = p.xxVel;
		yyVel = p.yyVel;
		mass = p.mass;
		imgFileName = p.imgFileName;
	}

	public double calcDistance(Planet p)
	{
		double dx_squared = (p.xxPos - xxPos) * (p.xxPos - xxPos);
		double dy_squared = (p.yyPos - yyPos) * (p.yyPos - yyPos);
		return Math.sqrt(dx_squared + dy_squared);
	}

	public double calcForceExertedBy(Planet p)
	{
		return Planet.gravity * mass * p.mass / (calcDistance(p) * calcDistance(p));
	}

	public double calcForceExertedByX (Planet p)
	{
		return calcForceExertedBy(p) * (p.xxPos - xxPos) / calcDistance(p);
	}

	public double calcForceExertedByY (Planet p)
	{
		return calcForceExertedBy(p) * (p.yyPos - yyPos) / calcDistance(p);
	}

	public double calcNetForceExertedByX (Planet[] p)
	{
		double total = 0;
		for (int i = 0; i < p.length; i++)
		{
			if (!equals(p[i]))
			{
				total += calcForceExertedByX(p[i]);
			}
		}

		return total;
	}

	public double calcNetForceExertedByY (Planet[] p)
	{
		double total = 0;
		for (int i = 0; i < p.length; i++)
		{
			if (!equals(p[i]))
			{
				total += calcForceExertedByY(p[i]);
			}
		}

		return total;
	}

	public void update(double dt, double fX, double fY)
	{
		double ax = fX / mass;
		double ay = fY / mass;
		xxVel += dt * ax;
		yyVel += dt * ay;
		xxPos += dt * xxVel;
		yyPos += dt * yyVel;
	}

	public void draw()
	{
		StdDraw.picture(xxPos, yyPos, "images/" + imgFileName);
	}
}
