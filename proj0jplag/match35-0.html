	public static Planet[] readPlanets(String file_name){

		double radius = in.readDouble();

public class Planet{
	// initialize instance variables
	public double xxPos;
	public double yyPos;
	public double xxVel;
	public double yyVel;
	public double mass;
	public String imgFileName;

	//First constructor
	public Planet(double xP, double yP, double xV, double yV, double m, String img){
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}

	//Second contructor. Makes a copy
	public Planet(Planet p){
		/*Not sure what to call the copy. Come back and check
		*Also check whether this only copies pointer or creates
		*new object */
		this.xxPos = p.xxPos;
		this.yyPos = p.yyPos;
		this.xxVel = p.xxVel;
		this.yyVel = p.yyVel;
		this.mass = p.mass;
		this.imgFileName = p.imgFileName;
	}

	public double calcDistance(Planet p){
		double x_dist = this.xxPos - p.xxPos;
		double y_dist = this.yyPos - p.yyPos;
		return Math.sqrt(x_dist*x_dist + y_dist*y_dist);
	}

	public double calcForceExertedBy(Planet p){
		double m1 = this.mass;
		double m2 = p.mass;
		double g = 6.67 * .00000000001;
		double r = this.calcDistance(p);
		return (g * m1 * m2) / (r * r);
	}

	public double calcForceExertedByX(Planet p){
		double x_dist =  p.xxPos - this.xxPos;
		double f = this.calcForceExertedBy(p);
		return f * x_dist / this.calcDistance(p);
	}
	public double calcForceExertedByY(Planet p){
		double y_dist = p.yyPos - this.yyPos;
		double f = this.calcForceExertedBy(p);
		return f * y_dist / this.calcDistance(p);
	}

	public double calcNetForceExertedByX(Planet[] allPlanets) {
		double x = 0.0;
		for(int i = 0; i < allPlanets.length; i++) {
			if(!allPlanets[i].equals(this)){
				x += this.calcForceExertedByX(allPlanets[i]);
			}
		}
		return x;
		
	}

	public double calcNetForceExertedByY(Planet[] allPlanets) {
		double y = 0.0;
		for(int i = 0; i < allPlanets.length; i++) {
			if(!allPlanets[i].equals(this)){
				y += this.calcForceExertedByY(allPlanets[i]);
			}
		}
		return y;
		
	}
	public void update(double dt, double x_f, double y_f) {
		double m = this.mass;
		double a_x = x_f / m; 
		double a_y = y_f / m;
		this.xxVel = this.xxVel + dt * a_x;
		this.yyVel = this.yyVel + dt * a_y;
		this.xxPos = this.xxPos + dt * this.xxVel;
		this.yyPos = this.yyPos + dt * this.yyVel;
	}

	public void draw(){
		StdDraw.picture(this.xxPos, this.yyPos, "images/" + this.imgFileName);
	}
}
