import java.util.Arrays;


public class NBody{ 

	public static double readRadius(String file) {

	In in = new In(file);
	int N = in.readInt();
	double R = in.readDouble();

	return R; 

	}

	public static Planet[] readPlanets(String file ){
	In in = new In(file);
	int N = in.readInt();
	double R = in.readDouble();
	Planet[] planets= new Planet[N];
	int counter = 0;
	while(!in.isEmpty() && counter < N ){

		double xxPos = in.readDouble();
	 	double yyPos = in.readDouble();
	 	double xxVel = in.readDouble();
	 	double yyVel = in.readDouble();
		double mass = in.readDouble();
		String imgFileName = in.readString();
		planets[counter] = new Planet(xxPos, yyPos, xxVel, yyVel, mass, imgFileName);
		counter++;

	}

	return planets;

	}


	public static void main(String[] args){ 
	double T = Double.parseDouble(args[0]);
	double dt = Double.parseDouble(args[1]);
	String filename = args[2];
	double radius = readRadius(filename);
	Planet[] planet_Arr = readPlanets(filename); 

	int n = planet_Arr.length;

	/* draw background*/		
	StdDraw.setScale(-radius, radius);
	StdDraw.picture(0, 0, "./images/starfield.jpg"); 


	
	for(int i = 0 ; i < planet_Arr.length; i++) {
		planet_Arr[i].draw();
		
	}
	


	double t = 0; 
	while(t <= T){
		Double[] xForces = new Double[n]; 
		Double[] yForces = new Double[n];

		for(int i = 0 ; i < planet_Arr.length; i++) {
			xForces[i] = planet_Arr[i].calcNetForceExertedByX(planet_Arr);
			yForces[i] = planet_Arr[i].calcNetForceExertedByY(planet_Arr);
			
		}

		for(int z = 0 ; z < planet_Arr.length; z++) {
			planet_Arr[z].update(dt, xForces[z], yForces[z]);
		}


		StdDraw.setScale(-radius, radius);
		StdDraw.picture(0, 0, "./images/starfield.jpg"); 

		for(int x = 0 ; x < planet_Arr.length; x++) {
			planet_Arr[x].draw();

		}

		StdDraw.show(10);
		t = t + dt;

	} 
	
		StdOut.printf("%d\n", planet_Arr.length);
		StdOut.printf("%.2e\n", radius);
		for (int i = 0; i < planet_Arr.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
   			planet_Arr[i].xxPos, planet_Arr[i].yyPos, planet_Arr[i].xxVel, planet_Arr[i].yyVel, planet_Arr[i].mass, planet_Arr[i].imgFileName);	
}		




	}


}

              double yV, double m, String img) {

		this.yyPos = yP;

		this.yyVel = yV;
		this.mass = m;
		this.imgFileName = img;

	}

		this.xxVel = p.xxVel;
		this.yyVel = p.yyVel ;
		this.mass = p.mass; 
		this.imgFileName = p.imgFileName; 

	}

	public double calcDistance(Planet a){
	 double x_dist =  (a.xxPos - xxPos)*(a.xxPos - xxPos);
	 double y_dist = (a.yyPos - yyPos)*(a.yyPos - yyPos);
	 double dist = Math.sqrt(x_dist + y_dist);

	 return dist;
	}


	public double calcForceExertedBy(Planet a){
		double dist = calcDistance(a);
		double G = 6.67e-11;
		double force = (G * (mass * a.mass))/ (dist * dist);
		return force;

	}

	public double calcForceExertedByX(Planet a){
		double f = calcForceExertedBy(a);
		double r = calcDistance(a);
		double x_dist = (a.xxPos - xxPos);

		double x_force = (f * x_dist)/r;
		return x_force;
	}


	public double calcForceExertedByY(Planet a){
		double f = calcForceExertedBy(a);
		double r = calcDistance(a);
		double y_dist = (a.yyPos - yyPos);

		double y_force = (f * y_dist)/r;
		return y_force;
	}


	public double calcNetForceExertedByX(Planet[] args){
		double net_force_x = 0;
		for (int i = 0; i < args.length; i++) {
			if (!this.equals(args[i])){
				net_force_x = net_force_x + calcForceExertedByX(args[i]);
			}


		}

		return net_force_x; 
	}

	public double calcNetForceExertedByY(Planet[] args){
		double net_force_y = 0;
		for (int i = 0; i < args.length; i++) {
			if (!this.equals(args[i])){
				net_force_y = net_force_y + calcForceExertedByY(args[i]);
			}
		}

		return net_force_y;

	}


	public void update(double dt, double fX, double fY){
		double x_accel = fX/mass;
		double y_accel = fY/mass;
		xxVel = xxVel + dt*x_accel;
		yyVel = yyVel + dt*y_accel;
		xxPos = xxPos + dt*xxVel;
		yyPos = yyPos + dt*yyVel;

	}

	public void draw() {
		StdDraw.picture(xxPos, yyPos, "images/" + imgFileName);
	}


}
