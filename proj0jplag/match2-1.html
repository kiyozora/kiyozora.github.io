public class NBody {
	public static double readRadius(String fileName) {
		In in = new In(fileName);
		in.readInt();
		return in.readDouble();
	}

	public static Planet[] readPlanets(String fileName) {
		In in = new In(fileName);
		int numberOfPlanets = in.readInt();
		double radius = in.readDouble();
		int planetNumber = 0;
		Planet[] planets = new Planet[numberOfPlanets];
		while (planetNumber < numberOfPlanets) {
			double xxPos = in.readDouble();
			double yyPos = in.readDouble();
			double xxVel = in.readDouble();
			double yyVel = in.readDouble();
			double mass = in.readDouble();
			String imgFileName = in.readString();
			Planet p = new Planet(xxPos, yyPos, xxVel, yyVel, mass, imgFileName);
			planets[planetNumber] = p;
			planetNumber = planetNumber + 1;
		}
		return planets;
	}


	public static void main(String[] args) {
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String filename = args[2];
		double radius = readRadius(filename);
		Planet[] planets = readPlanets(filename);
		StdDraw.setXscale(-radius, radius);
		StdDraw.setYscale(-radius, radius);	
		double time = 0;
		while (time < T) {
			double[] xForces = new double[planets.length];
			double[] yForces = new double[planets.length];
			StdDraw.picture(0, 0, "images/starfield.jpg");
			int index2 = 0;
			while (index2 < planets.length) {
				xForces[index2] = planets[index2].calcNetForceExertedByX(planets);
				yForces[index2] = planets[index2].calcNetForceExertedByY(planets);
				index2 = index2 + 1;
			}
			index2 = 0;
			while (index2 < planets.length) {
				planets[index2].update(dt, xForces[index2], yForces[index2]);
				// StdAudio.play("audio/2001.mid");
				index2 = index2 + 1;
			}
			int index1 = 0;
			while (index1 < planets.length) {
				planets[index1].draw();
				index1 = index1 + 1;
			}
			StdDraw.show(10);
			time = time + dt;
		}
		StdOut.printf("%d\n", planets.length);
		StdOut.printf("%.2e\n", radius);
		for (int i = 0; i < planets.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
   			planets[i].xxPos, planets[i].yyPos, planets[i].xxVel, planets[i].yyVel, planets[i].mass,
   			planets[i].imgFileName);	
		}
	}
}

public class Planet {
	public double xxPos; // Its current x position
	public double yyPos; // Its current y position
	public double xxVel; // Its current velocity in the x direction
	public double yyVel; // Its current velocity in the y direction
	public double mass; // Its mass
	public String imgFileName; /* The name of an image in the images
						 * directory that depicts the planet
						 */

	public Planet(double xP, double yP, double xV,
				  double yV, double m, String img) {
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}

	public Planet(Planet p) {
		xxPos = p.xxPos;
		yyPos = p.yyPos;
		xxVel = p.xxVel;
		yyVel = p.yyVel;
		mass = p.mass;
		imgFileName = p.imgFileName;
	}
	
	public double calcDistance(Planet p) {
		return Math.sqrt(Math.pow((this.xxPos - p.xxPos), 2) + Math.pow((this.yyPos - p.yyPos), 2));
	}

	public double calcForceExertedBy(Planet p) {
		return 6.67e-11 * (this.mass * p.mass) / Math.pow(this.calcDistance(p), 2);
	}

	public double calcForceExertedByX(Planet p) {
		return this.calcForceExertedBy(p) * (p.xxPos - this.xxPos) / this.calcDistance(p);
	}

	public double calcForceExertedByY(Planet p) {
		return this.calcForceExertedBy(p) * (p.yyPos - this.yyPos) / this.calcDistance(p);
	}

	public double calcNetForceExertedByX(Planet[] planetArray) {
		double netForce = 0;
		int index = 0;
		if (planetArray.length == 0) {
			return 0;
		} while (index < planetArray.length) {
			if (this.equals(planetArray[index])) {
				index = index + 1;
			} else {
				netForce = netForce + this.calcForceExertedByX(planetArray[index]);
				index = index + 1;
			}
		}
		return netForce;
	}

	public double calcNetForceExertedByY(Planet[] planetArray) {
		double netForce = 0;
		int index = 0;
		if (planetArray.length == 0) {
			return 0;
		} while (index < planetArray.length) {
			if (this.equals(planetArray[index])) {
				index = index + 1;
			} else {
				netForce = netForce + this.calcForceExertedByY(planetArray[index]);
				index = index + 1;
			}
		}
		return netForce;
	}

	public void update(double dt, double fX, double fY) {
		double xxAcc = fX / this.mass;
		double yyAcc = fY / this.mass;
		xxVel = xxVel + dt * xxAcc;
		yyVel = yyVel + dt * yyAcc;
		xxPos = xxPos + dt * xxVel;
		yyPos = yyPos + dt * yyVel;
	}

	public void draw() {
		StdDraw.picture(xxPos, yyPos, "images/" + imgFileName);
	}
