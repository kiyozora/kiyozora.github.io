/** Class that will simulate a universe specified in the data files. 
 * 
 *  @author Susan Shen 
 *  
 */

public class NBody { 
	
	/** Sets up the animation and runs through it. */ 
	public static void main(String[] args) {
		/** Parses the command line. */ 
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String fileName = args[2];
		
		double time = 0; 
		double universeRadius = readRadius(fileName);
		Planet[] allPlanets = readPlanets(fileName);
		StdDraw.setScale(-universeRadius, universeRadius);
		
		/** Draws the initial planet setup. */
		StdDraw.picture(0.0, 0.0, "images/starfield.jpg");
		for (Planet p : allPlanets) {
			p.draw();
		}
		
		/** Loop that runs through each increment of time and updates
		 * the display during each increment, with 10 millisecond delays. 
		 */
		while (time <= T) {
			double[] xForces = new double[allPlanets.length];
			double[] yForces = new double[allPlanets.length];
			
			for (int i = 0; i < allPlanets.length; i++) {
				xForces[i] = allPlanets[i].calcNetForceExertedByX(allPlanets);
				yForces[i] = allPlanets[i].calcNetForceExertedByY(allPlanets);
			}
			StdDraw.picture(0.0, 0.0, "images/starfield.jpg");
			for (int i = 0; i < allPlanets.length; i++) {
				allPlanets[i].update(dt, xForces[i], yForces[i]);
				allPlanets[i].draw();
			}
			StdDraw.show(10);
			time += dt;
		}
		
		/** Prints out the final position of the planets. */ 
		StdOut.printf("%d\n", allPlanets.length);
		StdOut.printf("%.2e\n", universeRadius);
		for (int i = 0; i < allPlanets.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
		   		allPlanets[i].xxPos, allPlanets[i].yyPos, allPlanets[i].xxVel, 
		   		allPlanets[i].yyVel, allPlanets[i].mass, allPlanets[i].imgFileName);
		}
	}
	
	/** Returns a double corresponding to the radius of the universe 
	 * in that file. */
	public static double readRadius(String file) {
		In in = new In(file);
		in.readInt();
		return in.readDouble();
	}
	
	/** Returns an array of Planets corresponding to the planets 
	 * in that file. */
	public static Planet[] readPlanets(String file) {
		In in = new In(file); 
		int number = in.readInt();
		in.readDouble();
		int counter = 0;
		Planet[] allPlanets = new Planet[number];
		while (counter != number) {
			double xxPos = in.readDouble();
			double yyPos = in.readDouble();
			double xxVel = in.readDouble();
			double yyVel = in.readDouble(); 
			double mass = in.readDouble();
			String imgFileName = in.readString();
			Planet p = new Planet(xxPos, yyPos, xxVel, yyVel, mass, imgFileName);
			allPlanets[counter] = p;
			counter += 1; 
		}
		return allPlanets;
	}
}

/** Public class Planet and its constructor
 * 
 * @author Susan Shen
 *
 */

public class Planet {
	public double xxPos; 
	public double yyPos;
	public double xxVel;
	public double yyVel;
	public double mass; 
	public String imgFileName;
	
	public Planet(double xp, double yP, double xV, 
			double yV, double m, String img) {
		xxPos = xp; 
		yyPos = yP;
		xxVel = xV; 
		yyVel = yV; 
		mass = m;
		imgFileName = img; 		
	}
	
	public Planet(Planet p){
		xxPos = p.xxPos;
		yyPos = p.yyPos;
		xxVel = p.xxVel;
		yyVel = p.yyVel;
		mass = p.mass;
		imgFileName = p.imgFileName;
	}
	
	/** Calculates the distance between two planets: the given planet
	 * and Planet p. */
	public double calcDistance(Planet p) { 
		double Pxxdiff = xxPos - p.xxPos; 
		double Pyydiff = yyPos - p.yyPos;
		return Math.sqrt((Pxxdiff * Pxxdiff) + (Pyydiff * Pyydiff));		
	}
	
	/** Calculates the force exerted on the given planet by Planet p. */
	public double calcForceExertedBy(Planet p) {
		double G = 6.67 * Math.pow(10, -11);
		double dx = xxPos - p.xxPos;
		double dy = yyPos - p.yyPos;
		double rsquared = (dx * dx) + (dy * dy);
		return (G * (p.mass * mass)) / rsquared;
	}
	
	/** Calculates the force exerted in the X direction on the given planet 
	 * by Planet p. */ 
	public double calcForceExertedByX(Planet p) {
		double force = calcForceExertedBy(p);
		double dx = p.xxPos - xxPos; 
		double dy = p.yyPos - yyPos;
		double rsquared = (dx * dx) + (dy* dy);
		double r = Math.sqrt(rsquared);
		return (force * dx) / r;
	}
	
	/** Calculates the force exerted in the Y direction on the given planet
	 * by Planet p. */
	public double calcForceExertedByY(Planet p) {
		double force = calcForceExertedBy(p);
		double dx = p.xxPos - xxPos; 
		double dy = p.yyPos - yyPos;
		double rsquared = (dx * dx) + (dy* dy);
		double r = Math.sqrt(rsquared);
		return (force * dy) / r;
	}
	
	/** Calculates the net force in the X direction on the given planet 
	 * by each planet in Array allPlanets. */
	public double calcNetForceExertedByX(Planet[] allPlanets) {
		double XNetForce = 0.0;
		for (Planet p : allPlanets) {
			double counter; 
			if (xxPos == p.xxPos && yyPos == p.yyPos) {
				counter = 0.0;
			} else {
				counter = calcForceExertedByX(p);
			}
			XNetForce += counter;
			counter = 0.0; 
		}
		return XNetForce;
	}
	
	/** Calculates the net force in the Y direction on the given planet 
	 * by each planet in Array allPlanets. */
	public double calcNetForceExertedByY(Planet[] allPlanets) {
		double YNetForce = 0.0;
		for (Planet p : allPlanets) {
			double counter; 
			if (xxPos == p.xxPos && yyPos == p.yyPos) {
				counter = 0.0;
			} else {
				counter = calcForceExertedByY(p);
			}
			YNetForce += counter;
			counter = 0.0; 
		}
		return YNetForce;
	}
	
	/** Determines how much the forces exerted on the planet will cause 
	 * the given planet to accelerate. */ 
	public void update(double dt, double fX, double fY) {
		double xAcceleration = fX / mass;
		double yAcceleration = fY / mass;
		xxVel += (dt * xAcceleration);
		yyVel += (dt * yAcceleration);
		xxPos += (dt * xxVel);
		yyPos += (dt * yyVel);
	}
	
	/** Draws the given planet. */
	public void draw() {
		StdDraw.picture(xxPos, yyPos, "images/" + imgFileName);
	}
}
