public class NBody{

    public static double readRadius(String file){
	In in = new In(file);
	int N = in.readInt();
	double radius = in.readDouble();
	return radius;
    }

    public static Planet[] readPlanets(String file){
	In in = new In(file);
	int N = in.readInt();
	double radius = in.readDouble();
     	int i = 0;
	Planet[] ps = new Planet[N];
	while (i < N) {
	    double x = in.readDouble();
	    double y = in.readDouble();
	    double vx = in.readDouble();
	    double vy = in.readDouble();
	    double m = in.readDouble();
	    String img = in.readString();
	    ps[i] = new Planet(x, y, vx, vy, m, img);
	    i += 1;
	}
	return ps;
    }

    public static void main(String[] args){
	StdAudio.play("./audio/2001.mid");
	double T = Double.parseDouble(args[0]);
	double dt = Double.parseDouble(args[1]);
	String filename = args[2];
	double r = readRadius(filename);
	Planet[] plst = readPlanets(filename);

	StdDraw.setScale(-r, r);
	StdDraw.picture(0, 0, "./images/starfield.jpg");

	int i = 0;
	while (i < plst.length) {
	    plst[i].draw();
	    i += 1;
	}

	int time = 0;

	while (time < T) {
	    double[] xForces = new double[plst.length];
	    double[] yForces = new double[plst.length];
	    int ind = 0;
	    while (ind < plst.length){
		double fx = plst[ind].calcNetForceExertedByX(plst);
		double fy = plst[ind].calcNetForceExertedByY(plst);
		xForces[ind] = fx;
		yForces[ind] = fy;
		plst[ind].update(dt, fx, fy);
		ind += 1;
	    }

	    StdDraw.picture(0, 0, "./images/starfield.jpg");
	    int j = 0;
	    while (j < plst.length) {
		plst[j].draw();
		j += 1;
	    }
	    StdDraw.show(10);
	    time += dt;
	}

	StdOut.printf("%d\n", plst.length);

Planet.java

public class Planet{
    public double xxPos;
    public double yyPos;
    public double xxVel;
    public double yyVel;
    public double mass;
    public String imgFileName;

    public Planet(double xP, double yP, double xV, double yV, double m, String img) {
	xxPos = xP;
	yyPos = yP;
	xxVel = xV;
	yyVel = yV;
	mass = m;
	imgFileName = img;
    }

   
    public Planet(Planet p) {
        xxPos = p.xxPos;
        yyPos = p.yyPos;
        xxVel = p.xxVel;
        yyVel = p.yyVel;
        mass = p.mass;
	imgFileName = p.imgFileName;

    }

    public double calcDistance(Planet p) {
	double dSquared = (xxPos - p.xxPos) * (xxPos - p.xxPos) + (yyPos - p.yyPos) * (yyPos - p.yyPos);
	double d = Math.sqrt(dSquared);
	return d;
    }

    public double calcForceExertedBy(Planet p) {
	double r = calcDistance(p);
	double f = 6.67 * Math.pow(10, -11) * mass * p.mass / (r * r);
	return f;
    }

    public double calcForceExertedByX(Planet p) {
	double r = calcDistance(p);
	double f = calcForceExertedBy(p);
	double fx = f * (p.xxPos - xxPos) / r;
	return fx;
    }

    public double calcForceExertedByY(Planet p) {
	double r = calcDistance(p);
	double f = calcForceExertedBy(p);
	double fy = f * (p.yyPos - yyPos) / r;
	return fy;
    }

    public double calcNetForceExertedByX(Planet[] ps){
	double xNetForce = 0;
	int n = ps.length;
	int i = 0;
	while (i < n){
	    if (this.equals(ps[i]) != true){
		xNetForce += calcForceExertedByX(ps[i]);
	    }
	    i += 1;
	}
	return xNetForce;
    }

    public double calcNetForceExertedByY(Planet[] ps){
	double yNetForce = 0;
	int n = ps.length;
	int i = 0;
	while (i < n){
	    if (this.equals(ps[i]) != true){
		yNetForce += calcForceExertedByY(ps[i]);
	    }
	    i += 1;
	}
	return yNetForce;
    }

    public void update(double dt, double fX, double fY){
	double ax = fX / mass;
	double ay = fY / mass;
	xxVel  = xxVel + ax * dt;
	yyVel  = yyVel + ay * dt;
	xxPos = xxPos + dt * xxVel;
	yyPos = yyPos + dt * yyVel;
    }

    public void draw(){
	StdDraw.picture(xxPos, yyPos, "./images/" + imgFileName);
    }
    
}
