public class NBody {
	public static double readRadius(String filePath) {
		In reader = new In(filePath);
		int size = reader.readInt();
		double radius = reader.readDouble();
		return radius;
	}

	public static Planet[] readPlanets(String filePath) {
		In reader = new In(filePath);
		int size = reader.readInt();
		double radius = reader.readDouble();
		Planet[] planets = new Planet[size];
		for (int i = 0; i < planets.length; i++) {
			double xPos = reader.readDouble();
			double yPos = reader.readDouble();
			double xVel = reader.readDouble();
			double yVel = reader.readDouble();
			double mass = reader.readDouble();
			String imgName = reader.readString();

			Planet tempPlanet = new Planet(xPos, yPos, 
											xVel, yVel,
											mass,imgName);
			planets[i] = tempPlanet;
		}
		return planets;
	}

	public static void main(String[] args) {
		//T, dt, filename of data "1231" "12312" "./data/planets.txt"
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String filename = args[2];
		double radius = readRadius(filename);
		Planet[] planets = readPlanets(filename);
		//Draws Background
		StdDraw.setScale((-1)*radius, radius);
		StdDraw.picture(0, 0, "./images/starfield.jpg");
		for (int i = 0; i < planets.length; i++) {
			planets[i].draw();
			/*System.out.println(planets[i].imgFileName);
			System.out.println(planets[i].xxPos);
			System.out.println(planets[i].yyPos);*/
		}
		double time = 0;
		
		while(time < T) {
			double[] xForces = new double[planets.length];
			for (int i = 0; i < planets.length; i++) {
				xForces[i] = planets[i].calcNetForceExertedByX(planets);
			}
			double[] yForces = new double[planets.length];
			for (int i = 0; i < planets.length; i++) {

				yForces[i] = planets[i].calcNetForceExertedByY(planets);
			}

			for (int i = 0; i < planets.length; i++) {
				planets[i].update(dt, xForces[i], yForces[i]);
			}

			StdDraw.picture(0, 0, "./images/starfield.jpg");

			for (int i = 0; i < planets.length; i++) {
				planets[i].draw();
			}

			StdDraw.show(10);
			time += dt;
		}
		StdOut.printf("%d\n", planets.length);
		StdOut.printf("%.2e\n", radius);
		for (int i = 0; i < planets.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
	   		planets[i].xxPos, planets[i].yyPos, planets[i].xxVel, planets[i].yyVel, planets[i].mass, planets[i].imgFileName);	
		}		

	}
}

public class Planet {
	private static final double gravConst = 6.67e-11;
	//should be 6.67e-11


	public double xxPos;
	public double yyPos;
	public double xxVel;
	public double yyVel;
	public double mass;
	public String imgFileName;

	//creates new planet

	public Planet(double xP, double yP, double xV, 
		double yV, double m, String img) {
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}

	//copies planet
	public Planet(Planet p) {
		xxPos = p.xxPos;
		yyPos = p.yyPos;
		xxVel = p.xxVel;
		yyVel = p.yyVel;
		mass = p.mass;
		imgFileName = p.imgFileName;
	}

	public double calcDistance(Planet p) {
		double xDistanceSquared = (p.xxPos - this.xxPos)*(p.xxPos - this.xxPos);
		double yDistanceSquared = (p.yyPos - this.yyPos)*(p.yyPos - this.yyPos);
		return Math.sqrt(xDistanceSquared + yDistanceSquared);
	}	

	public double calcForceExertedBy(Planet p) {
		//double gravConst = 6.67e-11;
		double distance = calcDistance(p);
		return gravConst*this.mass*p.mass/(distance*distance);

	}

	public double calcForceExertedByX(Planet p) {
		double distance = p.xxPos - this.xxPos;
		return calcForceExertedBy(p)*distance/calcDistance(p); 
	}

	public double calcForceExertedByY(Planet p) {
		double distance = p.yyPos - this.yyPos;
		return calcForceExertedBy(p)*distance/calcDistance(p); 
	}

	public double calcNetForceExertedByX(Planet[] planets) {
		double totalForce = 0;
		for (int i = 0; i < planets.length; i++) {
			if (!(planets[i].equals(this))) {
				totalForce += this.calcForceExertedByX(planets[i]);
			} 
		}
		return totalForce;
	}

	public double calcNetForceExertedByY(Planet[] planets) {
		double totalForce = 0;
		for (int i = 0; i < planets.length; i++) {
			if (!(planets[i].equals(this))) {
				totalForce += this.calcForceExertedByY(planets[i]);
			} 
		}
		return totalForce;
	}

	public void update(double dt, double fX, double fY) {

		xxVel = xxVel + dt*fX/mass;
		yyVel = yyVel + dt*fY/mass;
		xxPos = xxPos + dt*xxVel;
		yyPos = yyPos + dt*yyVel;
	}

	public void draw () {
		StdDraw.picture(xxPos, yyPos, "./images/" + imgFileName);
	}

}
