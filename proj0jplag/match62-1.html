NBody.java

public class NBody {
    public static void main(String[] args) {
        //Check command line arguments
        if(args.length != 3) {
            System.out.println("3 arguments required: T, dt, filename");
            return;
        }
        // Parse command line arguments

        double T = Double.parseDouble(args[0]);
        double dt = Double.parseDouble(args[1]);
        String filename = args[2];

        // Read information from provided file
        double radius = readRadius(filename);
        Planet[] planets = readPlanets(filename);

        // Set up the universe
        StdDraw.setScale(-radius, radius);
        StdDraw.clear();
        StdDraw.picture(0, 0, "images/starfield.jpg");

        for(Planet p : planets) {
            p.draw();
        }

        // Play music -- removed due to performance issues
        // StdAudio.play("audio/2001.mid");

        // Animate the universe
        for(double time = 0; time < T; time += dt) {
            double[] xForces = new double[planets.length];
            double[] yForces = new double[planets.length];

            // Calculate x and y forces for each planet
            for(int i = 0; i < planets.length; i++) {
                xForces[i] = planets[i].calcNetForceExertedByX(planets);
                yForces[i] = planets[i].calcNetForceExertedByY(planets);
            }

            // Update the planets
            for(int i = 0; i < planets.length; i++) {
                planets[i].update(dt, xForces[i], yForces[i]);
            }

            // Draw background image
            StdDraw.picture(0, 0, "images/starfield.jpg");

            // Draw the planets
            for(Planet p : planets) {
                p.draw();
            }
            
            StdDraw.show(10);
        }

        // Print final state of universe
        System.out.println(planets.length);
        System.out.println(radius);
        for(Planet p : planets) {
            System.out.print(p.xxPos + " " + p.yyPos + " ");
            System.out.print(p.xxVel + " " + p.yyVel + " ");
            System.out.println(p.mass + " " + p.imgFileName);
        }
    }


    public static double readRadius(String filename) {
        In in = new In(filename);
        in.readLine();
        return in.readDouble();
    }

    public static Planet[] readPlanets(String filename) {
        In in = new In(filename);
        int totalPlanets = in.readInt();
        Planet[] planets = new Planet[totalPlanets];

        // Skip the next number representing universe radius

        in.readDouble();

        for(int i = 0; i < totalPlanets; i++) {
            // Get planet information from file
            double xPos = in.readDouble();
            double yPos = in.readDouble();
            double xVel = in.readDouble();
            double yVel = in.readDouble();
            double mass = in.readDouble();
            String img = in.readString();

            // Store planet in array
            planets[i] = new Planet(xPos, yPos, xVel, yVel, mass, img);
        }
        return planets;
    }
}

Planet.java

public class Planet {
    public double xxPos;
    public double yyPos;
    public double xxVel;
    public double yyVel;
    public double mass;
    public String imgFileName;

    public Planet(double xP, double yP, double xV,
            double yV, double m, String img) {
        xxPos = xP;
        yyPos = yP;
        xxVel = xV;
        yyVel = yV;
        mass = m;
        imgFileName = img;
    }

    public Planet(Planet p) {
        this.xxPos = p.xxPos;
        this.yyPos = p.yyPos;
        this.xxVel = p.xxVel;
        this.yyVel = p.yyVel;
        this.mass = p.mass;
        this.imgFileName = p.imgFileName;
    }

    public double calcDistance(Planet p) {
        double dx = p.xxPos - this.xxPos;
        double dy = p.yyPos - this.yyPos;
        return Math.sqrt(dx*dx + dy*dy);
    }

    public double calcForceExertedBy(Planet p) {
        double G = 6.67e-11;

        double r = calcDistance(p);
        return G*this.mass*p.mass/(r*r);
    }

    public double calcForceExertedByX(Planet p) {
        double dx = p.xxPos - this.xxPos;
        double r = calcDistance(p);
        double force = calcForceExertedBy(p);
        return force*dx/r;
    }

    public double calcForceExertedByY(Planet p) {
        double dy = p.yyPos - this.yyPos;
        double r = calcDistance(p);
        double force = calcForceExertedBy(p);
        return force*dy/r;
    }

    public double calcNetForceExertedByX(Planet[] planets) {
        double netForceX = 0;
        for(Planet p : planets) {
            if(!this.equals(p)) {
                netForceX += calcForceExertedByX(p);
            }
        }
        return netForceX;
    }

    public double calcNetForceExertedByY(Planet[] planets) {
        double netForceY = 0;
        for(Planet p : planets) {
            if(!this.equals(p)) {
                netForceY += calcForceExertedByY(p);
            }
        }
        return netForceY;
    }

    public void update(double dt, double fX, double fY) {
        // Compute acceleration from applied forces
        double aX = fX/mass;
        double aY = fY/mass;

        // Calculate new velocity
        xxVel += aX*dt;
        yyVel += aY*dt;

        // Calculate new position
        xxPos += xxVel*dt;
        yyPos += yyVel*dt;
    }

    public void draw() {
        StdDraw.picture(xxPos, yyPos, "images/" + imgFileName);
    }
}
