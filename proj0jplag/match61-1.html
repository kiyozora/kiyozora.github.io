public class NBody {

	public static double readRadius(String fileName) {
		In fR = new In(fileName);
		fR.readLine();
		return fR.readDouble();//radius is on second line
	}

	public static Planet[] readPlanets(String fileName) {
		In fR = new In(fileName);
		Planet[] allPlanets = new Planet[fR.readInt()];
		int i = 0; //this keep track of where in the array we are
		fR.readDouble();

		while (i < allPlanets.length) {
			
			allPlanets[i] = new Planet(fR.readDouble(), fR.readDouble(), 
									fR.readDouble(), fR.readDouble(),
									fR.readDouble(), fR.readString());

			i++;
			fR.readLine();


		}
		return allPlanets;
	}

	public static void main(String[] args) {
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String fileName = args[2];
		double universeRadius = readRadius(fileName);
		Planet[] allPlanets = readPlanets(fileName);
		StdDraw.setScale(-universeRadius, universeRadius);
		
		StdDraw.picture(0, 0, 
						"images/starfield.jpg");
		StdDraw.show();
		for(Planet p: allPlanets) {
			p.draw();
		}

		double time = 0;
		while (time < T) {
			double xForces[] = new double[allPlanets.length];
			double yForces[] = new double[allPlanets.length];

			for (int i = 0; i < allPlanets.length; i++) {
				xForces[i] = allPlanets[i].calcNetForceExertedByX(allPlanets);
				yForces[i] = allPlanets[i].calcNetForceExertedByY(allPlanets);
			}
			for (int i = 0; i < allPlanets.length; i++) {
				allPlanets[i].update(dt, xForces[i], yForces[i]);
			}
			StdDraw.picture(0, 0, 
						"images/starfield.jpg");
			for(Planet p: allPlanets) {
				p.draw();
			}
			StdDraw.show(10);
			time += dt;
		}
		StdOut.printf("%d\n", allPlanets.length);
		StdOut.printf("%.2e\n", universeRadius);
		for (int i = 0; i < allPlanets.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
		   		allPlanets[i].xxPos, allPlanets[i].yyPos, allPlanets[i].xxVel, 
		   		allPlanets[i].yyVel, allPlanets[i].mass, allPlanets[i].imgFileName);	
}
	}
}

Planet.java

public class Planet {

	public double xxPos; //current x position
	public double yyPos; //current y position
	public double xxVel; //current x velocity
	public double yyVel; //current y velocity
	public double mass; //planet's mass
	public String imgFileName; //name of the image for the planet

	public Planet(double xP, double yP, double xV, double yV,
				  double m, String img) {
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}

	public Planet(Planet p) {
		xxPos = p.xxPos;
		yyPos = p.yyPos;
		xxVel = p.xxVel;
		yyVel = p.yyVel;
		mass = p.mass;
		imgFileName = p.imgFileName;
	}

	/*returns the distance between this planet and the planet passed
	 *as an argument*/
	public double calcDistance(Planet p) {
		double dx2 = Math.pow(xxPos - p.xxPos, 2);
		double dy2 = Math.pow(yyPos - p.yyPos, 2);

		return Math.pow(dx2 + dy2, .5);
	}

	//returns the force exerted on the argument planet by this planet
	public double calcForceExertedBy(Planet p) {

		double gravityConstant = 6.67 * Math.pow(10, -11);
		return mass*p.mass*gravityConstant/Math.pow(calcDistance(p), 2);
	}

	//returns the force in the x direction exerted by the argument planet
	public double calcForceExertedByX(Planet p) {
		return calcForceExertedBy(p)*(p.xxPos-xxPos)/calcDistance(p);
	}

	//returns the force in the y direction exerted by the argument planet
	public double calcForceExertedByY(Planet p) {
		return calcForceExertedBy(p)*(p.yyPos-yyPos)/calcDistance(p);
	}

	public double calcNetForceExertedByX(Planet[] allPlanets) {

		double totalForceX = 0;

		for (Planet p: allPlanets) {
			if (!p.equals(this)){
				totalForceX += calcForceExertedByX(p);
			}
		}
		return totalForceX;
	}

	public double calcNetForceExertedByY(Planet[] allPlanets) {

		double totalForceY = 0;

		for (Planet p: allPlanets) {
			if (!p.equals(this)) {
				totalForceY += calcForceExertedByY(p);
			}
		}
		return totalForceY;
	}

	public void update(double dt, double fX, double fY) {

		//calculates acceleration of planet
		double xAccl = fX/mass;
		double yAccl = fY/mass;

		//calculates new velocity
		xxVel += xAccl*dt;
		yyVel += yAccl*dt;

		//calculates new position
		xxPos += xxVel*dt;
		yyPos += yyVel*dt;
	}

	public void draw() {
		StdDraw.picture(xxPos, yyPos, imgFileName);
	}

}
