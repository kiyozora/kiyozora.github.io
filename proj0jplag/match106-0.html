public class NBody {
	private static int readNum(String fileName) {
		In scan = new In(fileName);
		return scan.readInt();
	}

	public static double readRadius(String fileName) {
		In radius = new In(fileName);
		radius.readInt();
		return radius.readDouble();
	}

	public static Planet[] readPlanets(String fileName) {
		In scan = new In(fileName);
		int number = scan.readInt();
		double radius = scan.readDouble();
		Planet[] now = new Planet[number];
		for(int i = 0; i < number; i++) {
			now[i] = new Planet(scan.readDouble(), scan.readDouble(), scan.readDouble(), scan.readDouble(), scan.readDouble(), scan.readString());
		}
		return now;
	}

	public static void main(String[] args) {
		double t = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String filename = args[2];
		int number = readNum(filename);
		double rad = readRadius(filename);
		Planet[] plans = readPlanets(filename);
		StdDraw.setXscale(-rad, rad);
		StdDraw.setYscale(-rad, rad);
		StdDraw.picture(0,0,"images/starfield.jpg");
		for(int i = 0; i < plans.length; i++){
			plans[i].draw();
		}
		StdAudio.play("audio/2001.mid");
		for(int time = 0; time <= t; time += dt){
			double[] xForces = new double[number];
			double[] yForces = new double[number];
			for(int j = 0; j < plans.length; j++){
				xForces[j] = plans[j].calcNetForceExertedByX(plans);
				yForces[j] = plans[j].calcNetForceExertedByY(plans);
			}
			for(int l = 0; l < plans.length; l++){
				plans[l].update(dt, xForces[l], yForces[l]);
			}
			StdDraw.picture(0,0,"images/starfield.jpg");
			for(int o = 0; o < plans.length; o++){
				plans[o].draw();
			}
			StdDraw.show(10);
		}
		StdOut.printf("%d\n", plans.length);
		StdOut.printf("%.2e\n", rad);
		for (int z = 0; z < plans.length; z++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
   			plans[z].xxPos, plans[z].yyPos, plans[z].xxVel, plans[z].yyVel, plans[z].mass, plans[z].imgFileName);
		}
	}
}

Planet.java

public class Planet {

	public double xxPos;
	public double yyPos;
	public double xxVel;
	public double yyVel;
	public double mass;
	public String imgFileName;
	private final double G = 6.67E-11;

	public Planet(double xP, double yP, double xV, double yV, double m, String img) {
	
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}

	public Planet(Planet p) {
		xxPos = p.xxPos;
		yyPos = p.yyPos;
		xxVel = p.xxVel;
		yyVel = p.yyVel;
		mass = p.mass;
		imgFileName = p.imgFileName;
	}

	public double calcDistance(Planet p){
		return Math.sqrt(((xxPos-p.xxPos)*(xxPos-p.xxPos)) + ((yyPos-p.yyPos)*(yyPos-p.yyPos)));
	}

	public double calcForceExertedBy(Planet p){
		return (G * mass * p.mass) / (calcDistance(p)*calcDistance(p));
	}

	public double calcForceExertedByX(Planet p){
		return calcForceExertedBy(p) * ((p.xxPos-xxPos)/calcDistance(p));
	}

	public double calcForceExertedByY(Planet p){
		return calcForceExertedBy(p) * ((p.yyPos-yyPos)/calcDistance(p));
	}

	public double calcNetForceExertedByX(Planet[] p){
		double net = 0;
		for(int i = 0; i < p.length; i++){
			if(!(equals(p[i]))){
				net += calcForceExertedByX(p[i]);
			}
		}
		return net;
	}

	public double calcNetForceExertedByY(Planet[] p){
		double net = 0;
		for(int i = 0; i < p.length; i++){
			if(!(equals(p[i]))){
				net += calcForceExertedByY(p[i]);
			}
		}
		return net;
	}

	public void update(double dt,double fX, double fY) {
		double aX = fX/mass;
		double aY = fY/mass;
		xxVel += dt*aX;
		yyVel += dt*aY;
		xxPos += dt*xxVel;
		yyPos += dt*yyVel;
	}

    public void draw() {
		StdDraw.picture(xxPos, yyPos, "/images/" + imgFileName);
    }
}
