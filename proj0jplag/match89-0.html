public class NBody {
	
	public static double readRadius(String filename) {
		In input = new In(filename);
		input.readInt();
		return input.readDouble();
	}

	public static Planet[] readPlanets(String filename) {
		In input = new In(filename);
		int length = input.readInt();
		input.readDouble();
		Planet[] planets = new Planet[length];
		for (int i = 0; i < length; i++) {
			Planet p = new Planet(input.readDouble(),input.readDouble(),input.readDouble(),input.readDouble(),input.readDouble(),input.readString());
			planets[i] = p;
		}
		input.close();
		return planets;
	}

	public static void main(String[] args) {
		// parse args
		double t = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String filename = args[2];
		double radius = readRadius(filename);
		Planet[] planets = readPlanets(filename);
		int numPlanets = planets.length;

		// play bgm
		StdAudio.play("audio/2001.mid");

		// draw universe
		StdDraw.setScale(-2*radius, 2*radius);
		StdDraw.clear();
		StdDraw.picture(0, 0, "images/starfield.jpg");
		for (int i = 0; i < numPlanets; i++) {
			planets[i].draw();
		}

		// animate universe
		double time = 0;
		while (time != t) {
			double[] xForces = new double[numPlanets];
			double[] yForces = new double[numPlanets];
			for (int i = 0; i < numPlanets; i++) {
				xForces[i] = planets[i].calcNetForceExertedByX(planets);
				yForces[i] = planets[i].calcNetForceExertedByY(planets);
			}
			for (int i = 0; i < numPlanets; i++) {
				planets[i].update(dt, xForces[i], yForces[i]);
			}
			StdDraw.picture(0, 0, "images/starfield.jpg");
			for (int i = 0; i < numPlanets; i++) {
				planets[i].draw();
			}
			StdDraw.show(10);
			time += dt;
		}

		// print universe
		StdOut.printf("%d\n", planets.length);
		StdOut.printf("%.2e\n", radius);
		for (int i = 0; i < planets.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
		   		planets[i].xxPos, planets[i].yyPos, planets[i].xxVel, planets[i].yyVel, planets[i].mass, planets[i].imgFileName);	
		}	
	}

}

Planet.java

public class Planet {
	
	public double xxPos;
	public double yyPos;
	public double xxVel;
	public double yyVel;
	public double mass;
	public String imgFileName;

	public Planet(double xP, double yP, double xV, double yV, double m, String img) {
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}

	public Planet(Planet P) {
		this.xxPos = P.xxPos;
		this.yyPos = P.yyPos;
		this.xxVel = P.xxVel;
		this.yyVel = P.yyVel;
		this.mass = P.mass;
		this.imgFileName = P.imgFileName;
	}

	///////////////////// calculate distance /////////////////////

	public double calcDistance(Planet p) {
		double xDiff = this.xxPos - p.xxPos;
		double yDiff = this.yyPos - p.yyPos;
		return Math.sqrt(xDiff * xDiff + yDiff * yDiff);
	}

	///////////////////// calculate force /////////////////////

	public double calcForceExertedBy(Planet p) {
		double gConstant = 6.67e-11;
		double distance = calcDistance(p);
		return gConstant * this.mass * p.mass / (distance * distance);
	}

	public double calcForceExertedByX(Planet p) {
		double force = calcForceExertedBy(p);
		double xDist = p.xxPos - this.xxPos;
		double distance = calcDistance(p);
		return force * xDist / distance;
	}

	public double calcForceExertedByY(Planet p) {
		double force = calcForceExertedBy(p);
		double yDist = p.yyPos - this.yyPos;
		double distance = calcDistance(p);
		return force * yDist / distance;
	}

	///////////////////// calculate net force /////////////////////

	public double calcNetForceExertedByX(Planet[] planets) {
		double total = 0;
		for (int i=0; i<planets.length; i++) {
			Planet p = planets[i];
			if (this.equals(p) != true) {
				total += calcForceExertedByX(p);
			}
		}
		return total;
	}

	public double calcNetForceExertedByY(Planet[] planets) {
		double total = 0;
		for (int i=0; i<planets.length; i++) {
			Planet p = planets[i];
			if (this.equals(p) != true) {
				total += calcForceExertedByY(p);
			}
		}
		return total;
	}

	///////////////////// update /////////////////////

	public void update(double dt, double xForce, double yForce) {
		double xAccel = xForce / mass;
		double yAccel = yForce / mass;
		xxVel += dt * xAccel;
		yyVel += dt * yAccel;
		xxPos += dt * xxVel;
		yyPos += dt * yyVel;
	}

	///////////////////// draw /////////////////////

	public void draw() {
		StdDraw.picture(xxPos, yyPos, "images/"+imgFileName);
	}

}
