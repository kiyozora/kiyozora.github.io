public class NBody{
	public static double readRadius(String data) {
		In in = new In(data);
		int num = in.readInt();
		double radius = in.readDouble();
		return radius;
	}

	public static Planet[] readPlanets(String data) {
		In in = new In(data);
		int num = in.readInt();
		Planet[] allPlanets = new Planet[num];
		double radius = in.readDouble();
		for (int i=0; i<num; i++) {
			double xP = in.readDouble();
			double yP = in.readDouble();
			double xV = in.readDouble();
			double yV = in.readDouble();
			double m = in.readDouble();
			String img = in.readString();
			allPlanets[i]= new Planet(xP, yP, xV, yV, m, img);
		}
		return allPlanets;
	}

	public static void main (String[] args) {
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String filename = args[2];
		Planet[] allPlanets = readPlanets(filename);
		StdAudio.play("audio/2001.mid");

		double size = readRadius(filename);
		String background = "images/starfield.jpg";
		StdDraw.setScale(-size, size);
		StdDraw.picture(0, 0, background, 2*size, 2*size);
		for (Planet p: allPlanets) {
			p.draw();
		}

		double timer = 0;
		while (timer <= T) {
			double[] xForces = new double[allPlanets.length];
			double[] yForces = new double[allPlanets.length];
			for (int i = 0; i<allPlanets.length; i++) {
				Planet p = allPlanets[i];
				xForces[i] = p.calcNetForceExertedByX(allPlanets);
				yForces[i] = p.calcNetForceExertedByY(allPlanets);
			}
			for (int i = 0; i<allPlanets.length; i++) {
				allPlanets[i].update(dt, xForces[i], yForces[i]);
			}

			StdDraw.setScale(-size, size);
			StdDraw.picture(0, 0, background, 2*size, 2*size);
			for (Planet p: allPlanets) {
				p.draw();
			}
			StdDraw.show(10);
			timer = timer + dt;	
		}

		StdOut.printf("%d\n", allPlanets.length);
		StdOut.printf("%.2e\n", size);
		for (int i = 0; i < allPlanets.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
		   		allPlanets[i].xxPos, allPlanets[i].yyPos, allPlanets[i].xxVel, allPlanets[i].yyVel, allPlanets[i].mass, allPlanets[i].imgFileName);	
		}		
	}
}

public class Planet {
	public double xxPos, yyPos, xxVel, yyVel, mass;
	public String imgFileName;
	

	public Planet(double xP, double yP, double xV, double yV, double m, String img) {
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}

	public Planet(Planet p) {
		xxPos = p.xxPos;
		yyPos = p.yyPos;
		xxVel = p.xxVel;
		yyVel = p.yyVel;
		mass = p.mass;
		imgFileName = p.imgFileName;
	}

	public double calcDistance(Planet p) {
		double xDis = xxPos - p.xxPos;
		double yDis = yyPos - p.yyPos;
		return Math.sqrt(xDis*xDis + yDis*yDis);
	}

	public double calcForceExertedBy(Planet p) {
		double g = 6.67 * Math.pow(10, -11);
		double dis = calcDistance(p);
		return g*mass*p.mass/dis/dis;
	}

	public double calcForceExertedByX(Planet p) {
		double force = calcForceExertedBy(p);
		double xDis = p.xxPos - xxPos;
		double dis = calcDistance(p);
		return force*xDis/dis;
	}

	public double calcForceExertedByY(Planet p) {
		double force = calcForceExertedBy(p);
		double yDis = p.yyPos - yyPos;
		double dis = calcDistance(p);
		return force*yDis/dis;
	}

	public double calcNetForceExertedByX(Planet[] allPlanets) {
		double forceX = 0;
		for (Planet p: allPlanets) {
			if (!p.equals(this)) {
				forceX = forceX + calcForceExertedByX(p);
			}
		}
		return forceX;
	}
	
	public double calcNetForceExertedByY(Planet[] allPlanets) {
		double forceY = 0;
		for (Planet p: allPlanets) {
			if (!p.equals(this)) {
				forceY = forceY + calcForceExertedByY(p);
			}
		}
		return forceY;
	}

	public void update(double dt, double xForce, double yForce) {
		double xAcc = xForce/mass;
		double yAcc = yForce/mass;
		xxVel = xxVel + xAcc*dt;
		yyVel = yyVel + yAcc*dt;
		xxPos = xxPos + dt*xxVel;
		yyPos = yyPos + dt*yyVel;
	}

	public void draw() {
		StdDraw.picture(xxPos, yyPos, "images/"+imgFileName);
	}

}
