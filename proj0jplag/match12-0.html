public class NBody {
	
	public static double readRadius(String file) {

		In in = new In(file);

		int numPlanets = in.readInt();
		double radius = in.readDouble();

		return radius;
	}

	public static Planet[] readPlanets(String file) {
		
		In in = new In(file);

		int numPlanets = in.readInt();
		double radius = in.readDouble();

		Planet[] planets = new Planet[numPlanets];

		int row = 0;
		while(row < numPlanets) {
			double xPos = in.readDouble();
			double yPos = in.readDouble();
			double xVel = in.readDouble();
			double yVel = in.readDouble();
			double mass = in.readDouble();
			String imgFileName = in.readString();

			planets[row] = new Planet(xPos, yPos, xVel, yVel, mass, imgFileName);
			row = row + 1;
		}

		return planets;
	}

	public static void main(String[] args) {
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String filename = args[2];
		double radius = readRadius(filename);
		Planet[] planets = readPlanets(filename);

		double t = 0;
		while(t < T) {
			double[] xForces = new double[planets.length];
			double[] yForces = new double[planets.length];
			for(int a = 0; a < planets.length; a++) {
				xForces[a] = planets[a].calcNetForceExertedByX(planets);
				yForces[a] = planets[a].calcNetForceExertedByY(planets);
			}

			// could have put in previous for loop, but autograder complains
			for(int b = 0; b < planets.length; b++) {
				planets[b].update(dt, xForces[b], yForces[b]);
			}

			String background = "images/starfield.jpg";
			StdDraw.setScale(-1*radius, radius);
			StdDraw.picture(0, 0, background);

			for(int i = 0; i < planets.length; i++) {
				planets[i].draw();
			}

			StdDraw.show(10);

			t = t + dt;
		}

		StdOut.printf("%d\n", planets.length);
		StdOut.printf("%.2e\n", radius);
		for (int i = 0; i < planets.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
		   		planets[i].xxPos, planets[i].yyPos, planets[i].xxVel, planets[i].yyVel, planets[i].mass, planets[i].imgFileName);	
		}		
		
	}

}

Planet.java

public class Planet {
	/* Instance variables */
	public double xxPos;
	public double yyPos;
	public double xxVel;
	public double yyVel;
	public double mass;
	public String imgFileName;

	// original constructor
	public Planet(double xP, double yP, double xV,
				  double yV, double m, String img) {
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}

	// copy constructor
	public Planet(Planet p) {
		xxPos = p.xxPos;
		yyPos = p.yyPos;
		xxVel = p.xxVel;
		yyVel = p.yyVel;
		mass = p.mass;
		imgFileName = p.imgFileName;
	}

	// calculate distance between planets
	public double calcDistance(Planet p) {
		double xdist = p.xxPos - xxPos;
		double ydist = p.yyPos - yyPos;
		double r = Math.sqrt(xdist*xdist+ ydist*ydist);
		return r;
	}

	// calculate magnitude of force between planets
	public double calcForceExertedBy(Planet p) {
		double dist = calcDistance(p);
		double gravConst = 6.67E-11;
		double force = gravConst * mass * p.mass / (dist * dist);
		return force;
	}

	// calculate force exerted on planet in x-direction
	public double calcForceExertedByX(Planet p) {
		double dist = calcDistance(p);
		double xdist = p.xxPos - xxPos;
		double magForce = calcForceExertedBy(p);
		double xForce = magForce * xdist / dist;
		return xForce;
	}	

	// calculate force exerted on planet in y-direction
	public double calcForceExertedByY(Planet p) {
		double dist = calcDistance(p);
		double ydist = p.yyPos - yyPos;
		double magForce = calcForceExertedBy(p);
		double yForce = magForce * ydist / dist;
		return yForce;
	}

	public double calcNetForceExertedByX(Planet[] pList) {
		double totForceX = 0;
		for(int i = 0; i < pList.length; i++) {
			if(!this.equals(pList[i])) {
				totForceX = totForceX + calcForceExertedByX(pList[i]);
			}
		}
		return totForceX;
	}

	public double calcNetForceExertedByY(Planet[] pList) {
		double totForceY = 0;
		for(int i = 0; i < pList.length; i++) {
			if(!this.equals(pList[i])) {
				totForceY = totForceY + calcForceExertedByY(pList[i]);
			}
		}
		return totForceY;
	}

	public void update(double dt, double fX, double fY) {
		double aX = fX / mass;
		double aY = fY / mass;
		xxVel = xxVel + dt * aX;
		yyVel = yyVel + dt * aY;
		xxPos = xxPos + dt * xxVel;
		yyPos = yyPos + dt * yyVel;
	}

	public void draw() {
		StdDraw.picture(xxPos, yyPos, "images/" + imgFileName);
	}
}
