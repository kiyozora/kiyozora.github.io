public class NBody {
	public static double readRadius(String file){
		In in = new In(file);
		String firstItem = in.readString();
		String secondItem = in.readString();	
		double radius = Double.parseDouble(secondItem);
		return radius;
	}

	public static Planet[] readPlanets(String file){
		In in = new In(file);
		String firstItem = in.readString();
		int planetNum = Integer.parseInt(firstItem);
		in.readString();
		Planet[] planets = new Planet[planetNum];
		for(int i =0; i < planetNum; i++){
			double xPos = in.readDouble();
			double yPos = in.readDouble();
			double xVel  = in.readDouble();
			double yVel = in.readDouble();
			double mass = in.readDouble();
			String imgName = in.readString();
			Planet p = new Planet(xPos, yPos, xVel, yVel, mass, imgName);
			planets[i] =p;
		}
		return planets;
	}

        if (args.length < 1) {

		try {

public class Planet {
	double xxPos;
	double yyPos;
	double xxVel;
	double yyVel;
	double mass;
	String imgFileName;

	public Planet(double xP, double yP, double xV, double yV, double m, String img) {
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}

	public Planet(Planet p) {
		this(p.xxPos, p.yyPos, p.xxVel, p.yyVel, p.mass, p.imgFileName);
	}

	public void draw() {
		StdDraw.picture(xxPos, yyPos, "./images/"+imgFileName);
	}

	public void update(double dt, double fX, double fY) {
		double anetX = fX / this.mass;
		double anetY = fY / this.mass;
		
		this.xxVel = this.xxVel + (dt * anetX);
		this.yyVel = this.yyVel + (dt * anetY);

		this.xxPos = this.xxPos + (dt * this.xxVel);
		this.yyPos = this.yyPos + (dt * this.yyVel);
	}
	
	public double calcDistance(Planet a){
		double dist = Math.sqrt(((this.xxPos - a.xxPos) * (this.xxPos - a.xxPos)) + ((this.yyPos - a.yyPos) * (this.yyPos - a.yyPos)));
		return dist;
	}
	
	public double calcForceExertedBy(Planet a){
		double force = (6.67*Math.pow(10,-11)) * this.mass * a.mass / Math.pow(this.calcDistance(a),2);
		return force;
	}
	
	public double calcForceExertedByX(Planet a){
		double force = this.calcForceExertedBy(a) * (a.xxPos - this.xxPos) / this.calcDistance(a);
		return force; 
	}
	
	public double calcForceExertedByY(Planet a){
		double force = this.calcForceExertedBy(a) * (a.yyPos - this.yyPos) / this.calcDistance(a);
		return force;
	}
	
	public double calcNetForceExertedByX(Planet[] allPlanets){
		double netForce = 0;
		int  i = 0;
		while(i < allPlanets.length){
			if (this.equals(allPlanets[i])){
				i += 1;
			}
			else{
				netForce = netForce + this.calcForceExertedByX(allPlanets[i]);
				i += 1;
			}
		}
		return netForce;			
	}	
	public double calcNetForceExertedByY(Planet[] allPlanets){
		double netForce = 0;
		int  i = 0;
		while(i < allPlanets.length){
			if (this.equals(allPlanets[i])){
				i += 1;
			}
			else{
				netForce = netForce + this.calcForceExertedByY(allPlanets[i]);
				i += 1;
			}
		}
		return netForce;			
	}
}
