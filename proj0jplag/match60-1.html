NBody.java

public class NBody {
  //class that runs the simulation
  public static double readRadius(String fileNameR) {
    //returns the radius of the universe
    In in = new In(fileNameR);
    int numPlanets = in.readInt();
    double radius = in.readDouble();
    return radius;
  }

  public static Planet[] readPlanets(String fileNameP) {
    //returns an array of Planets
    In in = new In(fileNameP);
    int numPlanets = in.readInt();
    double radius = in.readDouble();

    Planet[] planets = new Planet[numPlanets]; //create array of planets
    for (int i=0; i<numPlanets; i++) {
      double xP = in.readDouble();
      double yP = in.readDouble();
      double xV = in.readDouble();
      double yV = in.readDouble();
      double m = in.readDouble();
      String img = in.readString();
      planets[i] = new Planet(xP, yP, xV, yV, m, img);
    }
    return planets;
  }

  public static void main(String[] args) {
    //main method for NBody
    StdAudio.play("./audio/2001.mid"); //play the theme to 2001: A Space Odyssey


    double T = Double.parseDouble(args[0]);
    double dt = Double.parseDouble(args[1]);
    String filename = args[2];
    double radius = readRadius(filename);
    Planet[] planets = readPlanets(filename); //array of planets


    StdDraw.setScale(-radius, radius);
    StdDraw.picture(0, 0, "./images/starfield.jpg");
    for (Planet planet : planets) {
      planet.draw();
    }

    double time = 0;
    while (time < T) {
      Double[] xForces = new Double[planets.length];
      Double[] yForces = new Double[planets.length];
      for (int i=0; i<planets.length; i++) {
        //calculate net forces, then store in force arrays
        xForces[i] = planets[i].calcNetForceExertedByX(planets);
        yForces[i] = planets[i].calcNetForceExertedByY(planets);
      }
      for (int i=0; i<planets.length; i++) {
        //update each planet's position, velocity, and acceleration
        planets[i].update(dt, xForces[i], yForces[i]);
      }
      StdDraw.picture(0, 0, "./images/starfield.jpg"); //draw background
      for (Planet planet : planets) {
        //draw each planet with updates
        planet.draw();
      }
      StdDraw.show(10); //10ms delay
      time += dt; //increment time
    }
    //print out the final state of the universe
    StdOut.printf("%d\n", planets.length);
    StdOut.printf("%.2e\n", radius);
    for (int i = 0; i < planets.length; i++) {
      StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
      planets[i].xxPos, planets[i].yyPos, planets[i].xxVel, planets[i].yyVel,
      planets[i].mass, planets[i].imgFileName);
    }
  }
}

Planet.java

public class Planet {
    public double xxPos;
    public double yyPos;
    public double xxVel;
    public double yyVel;
    public double mass;
    public String imgFileName;

    public Planet(double xP, double yP, double xV, double yV, double m, String img) {
      xxPos = xP;
      yyPos = yP;
      xxVel = xV;
      yyVel = yV;
      mass = m;
      imgFileName = img;
    }

    public Planet(Planet p) {
      //constructs a duplicate of p
      xxPos = p.xxPos;
      yyPos = p.yyPos;
      xxVel = p.xxVel;
      yyVel = p.yyVel;
      mass = p.mass;
      imgFileName = p.imgFileName;
    }

    public double calcDistance(Planet p) {
      //returns distance between planets
      double dx = xxPos - p.xxPos;
      double dy = yyPos - p.yyPos;
      return Math.sqrt(dx*dx + dy*dy);
    }

    public double calcForceExertedBy(Planet p) {
      //returns force exerted by p
      double g = 6.67 * Math.pow(10, -11);

      double r = calcDistance(p);
      return g * mass * p.mass / (r*r);
    }

    public double calcForceExertedByX(Planet p) {
      //returns force exerted by p in the X direction
      double f = calcForceExertedBy(p);
      double dx = p.xxPos - xxPos;
      double r = calcDistance(p);
      return f * dx / r;
    }

    public double calcForceExertedByY(Planet p) {
      //returns force exerted by p in the Y direction
      double f = calcForceExertedBy(p);
      double dy = p.yyPos - yyPos;

      double r = calcDistance(p);
      return f * dy / r;
    }

    public double calcNetForceExertedByX(Planet[] allPlanets) {
      //returns *net* force exerted by p in the X direction
      double netForceX = 0;
      for (Planet p : allPlanets) {
        if (this.equals(p)) {
          //ignore same planet
          netForceX += 0;
        } else {

          netForceX += calcForceExertedByX(p);
        }
      }
      return netForceX;
    }

    public double calcNetForceExertedByY(Planet[] allPlanets) {
      //returns *net* force exerted by p in the Y direction
      double netForceY = 0;
      for (Planet p : allPlanets) {
        if (this.equals(p)) {
          //ignore same planet
          netForceY += 0;
        } else {

          netForceY += calcForceExertedByY(p);
        }
      }
      return netForceY;
    }

    public void update(double dt, double fX, double fY) {
      //calculates new velocity and position after dt
      double aX = fX / mass;
      double aY = fY / mass;
      xxVel = xxVel + dt * aX;
      yyVel = yyVel + dt * aY;
      xxPos = xxPos + dt * xxVel;
      yyPos = yyPos + dt * yyVel;
    }

    public void draw() {
      StdDraw.picture(xxPos, yyPos, "./images/"+imgFileName);
    }

}
