public class NBody {
	public static double readRadius(String fileName) {     //reads and returns the planet radius R in file fileName
		In file = new In(fileName);
		int N = file.readInt();     //reads the first item (int) of fileName and sets the value equal to N
		double R = file.readDouble();     //reads the second item (double) of fileName and sets the value equal to R
		return R;
	}

	public static Planet[] readPlanets(String fileName) {     //reads and returns an array of planets allPlanets in file fileName
		In file = new In(fileName);
		int N = file.readInt();     //see above method
		double R = file.readDouble();     //see above method
		Planet[] allPlanets = new Planet[N];     //creates a ne array of planets allPlanets and sets its size equal to N
		for (int p = 0; p < N; p++) {
			double pXPos = file.readDouble();
			double pYPos = file.readDouble();
			double pXVel = file.readDouble();
			double pYVel = file.readDouble();
			double pMass = file.readDouble();
			String pName = file.readString();
			allPlanets[p] = new Planet(pXPos, pYPos, pXVel, pYVel, pMass, pName);     //sets item p of allPlanets equal to a new planet containing the information gathered above
			//citation: http://www.itcsolutions.eu/2010/12/29/tutorial-java-4-1-how-to-add-a-new-element-to-a-java-array/
		}
		return allPlanets;
	}

	public static void main(String[] args) {     //creates a solar system (with sound!)
		double T = Double.parseDouble(args[0]);     //sets T equal to a String converted into a double
		double dt = Double.parseDouble(args[1]);    //sets dt equal to a String converted into a double
		//citation: http://stackoverflow.com/questions/5769669/convert-string-to-double-in-java
		String filename = args[2];
		double R = NBody.readRadius(filename);     //see above method
		Planet[] allPlanets = NBody.readPlanets(filename);     //calls readPlanets on filename
		StdDraw.setScale(-1 * R, R);     //sets the canvas size from -R to +R
		StdDraw.picture(0, 0, "images/starfield.jpg");     //sets the picture "images/starfield.jpg" in position 0, 0
		for (int p = 0; p < allPlanets.length; p++) {
			allPlanets[p].draw();     //draws a planet
		}
		double time = 0.0;
		StdAudio.play("audio/2001.mid");     //plays the soundtrack
		//citation: http://stackoverflow.com/questions/7875861/multiple-sounds-on-java
		while (time < T) {
			double[] xForces = new double[allPlanets.length];
			double[] yForces = new double[allPlanets.length];
			for (int p = 0; p < allPlanets.length; p++) {
				xForces[p] = allPlanets[p].calcNetForceExertedByX(allPlanets);
				yForces[p] = allPlanets[p].calcNetForceExertedByY(allPlanets);
			}
			for (int p1 = 0; p1 < allPlanets.length; p1++) {
				allPlanets[p1].update(dt, xForces[p1], yForces[p1]);     //updates the instance variables of each planet
			}
			StdDraw.clear();     //clears the canvas
			StdDraw.picture(0, 0, "images/starfield.jpg");
			for (int p2 = 0; p2 < allPlanets.length; p2++) {
				allPlanets[p2].draw();
			}
			StdDraw.show(10);
			time = time + dt;
		}
		System.out.println(allPlanets.length);
		System.out.println(R);
		for (int plan = 0; plan < allPlanets.length; plan++) {
			System.out.println(allPlanets[plan].xxPos + " " + allPlanets[plan].yyPos + " " + allPlanets[plan].xxVel + " " + allPlanets[plan].yyVel + " " + allPlanets[plan].mass + " " + allPlanets[plan].imgFileName);
		}
	}
}

public class Planet {
	public double xxPos;
	public double yyPos;
	public double xxVel;
	public double yyVel;
	public double mass;
	public String imgFileName;

	public Planet(double xP, double yP, double xV,
              double yV, double m, String img) {     //creates the first method of instantiating a planet
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}
	public Planet(Planet p) {     //creates the second method if instanyiating a planet
		xxPos = p.xxPos;
		yyPos = p.yyPos;
		xxVel = p.xxVel;
		yyVel = p.yyVel;
		mass = p.mass;
		imgFileName = p.imgFileName;
	}

	public double calcDistance(Planet p) {     //calculates teh distance between two planets
		double xxDifSq = (this.xxPos - p.xxPos) * (this.xxPos - p.xxPos);
		double yyDifSq = (this.yyPos - p.yyPos) * (this.yyPos - p.yyPos);
		return Math.pow(xxDifSq + yyDifSq, 0.5);
	}
	public double calcForceExertedBy(Planet p) {     //calculates the force two planets exerts on each other
		return 6.67e-11 * this.mass * p.mass / (this.calcDistance(p) * this.calcDistance(p));
	}
	public double calcForceExertedByX(Planet p) {     //calculates the force exerted in the x-axis
		return this.calcForceExertedBy(p) * (p.xxPos - this.xxPos) / this.calcDistance(p);
	}
	public double calcForceExertedByY(Planet p) {     //calculates the force exerted in the y-axis
		return this.calcForceExertedBy(p) * (p.yyPos - this.yyPos) / this.calcDistance(p);
	}
	public double calcNetForceExertedByX(Planet[] allPlanets) {     //calculates the net force exerted among planets in the x-axis
		double netForceXX = 0.0;
		for (int p = 0; p < allPlanets.length; p++) {
			if (!this.equals(allPlanets[p])) {
				netForceXX += this.calcForceExertedByX(allPlanets[p]);
			}
		}
		return netForceXX;
	}
	public double calcNetForceExertedByY(Planet[] allPlanets) {     //calculates the net force exerted among planets in the y-axis
		double netForceYY = 0.0;
		for (int p = 0; p < allPlanets.length; p++) {
			if (!this.equals(allPlanets[p])) {
				netForceYY += this.calcForceExertedByY(allPlanets[p]);
			}
		}
		return netForceYY;
	}
	public void update(double dt, double fX, double fY) {     //updates a planet's instance variables
		double accNetXX = fX / this.mass;
		double accNetYY = fY / this.mass;
		this.xxVel = this.xxVel + dt * accNetXX;
		this.yyVel = this.yyVel + dt * accNetYY;
		this.xxPos = this.xxPos + dt * this.xxVel;
		this.yyPos = this.yyPos + dt * this.yyVel;
	}

	public void draw() {     //draws a planet using its position
		StdDraw.picture(this.xxPos, this.yyPos, "images/" + this.imgFileName);
	}
}
