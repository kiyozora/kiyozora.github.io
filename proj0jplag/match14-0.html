public class NBody {
	
	public static double readRadius(String file) {
		In in = new In(file);
		double radius = 1.0;
		in.readInt();
		return in.readDouble();
	}

	public static Planet[] readPlanets(String file) {
		In in = new In(file);
		int numplanets = in.readInt();
		in.readDouble();
		Planet[] allPlanets = new Planet[numplanets];
		for (int i = 0; i < numplanets; i++) {
			allPlanets[i] = new Planet(in.readDouble(), in.readDouble(), in.readDouble(), in.readDouble(), in.readDouble(), in.readString());
		}
		return allPlanets;
	}

	public static void main(String[] args) {
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String filename = args[2];
		Planet[] p = readPlanets(filename);
		double r = readRadius(filename);
		StdDraw.setScale(-r, r);
		StdDraw.picture(0.0, 0.0, "images/starfield.jpg");
		int numplanets = p.length;
		int i = 0;
		double time = 0;
		int waitTimeMilliseconds = 10;

		while (i < numplanets) {
			StdDraw.picture(p[i].xxPos, p[i].yyPos, "images/" + p[i].imgFileName);
			i = i + 1;
		}

		while (time < T) {
			double[] xForces = new double[numplanets];
			double[] yForces = new double[numplanets];
			
			for (i = 0; i < numplanets; i++) {
				xForces[i] = p[i].calcNetForceExertedByX(p);
				yForces[i] = p[i].calcNetForceExertedByY(p);
			}

			StdDraw.picture(0.0, 0.0, "images/starfield.jpg");
			
			for (i = 0; i < numplanets; i++) {
				StdDraw.picture(p[i].xxPos, p[i].yyPos, "images/" + p[i].imgFileName);
			}

			for (i = 0; i < numplanets; i++) {
				p[i].update(dt, xForces[i], yForces[i]);
			}
			StdDraw.show(waitTimeMilliseconds);
			time = time + dt;
		}

		StdOut.printf("%d\n", p.length);
		StdOut.printf("%.2e\n", r);
		for (i = 0; i < p.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n", p[i].xxPos, p[i].yyPos, p[i].xxVel, p[i].yyVel, p[i].mass, p[i].imgFileName);
		}		
	}
}

Planet.java

public class Planet {
	
	public double xxPos;
	public double yyPos;
	public double xxVel;
	public double yyVel;
	public double mass;
	public String imgFileName;

	public Planet(double xP, double yP, double xV, double yV, double m, String img) {
		xxPos = xP;
		yyPos = yP;
		xxVel = xV;
		yyVel = yV;
		mass = m;
		imgFileName = img;
	}
	
	public Planet(Planet p) {
		xxPos = p.xxPos;
		yyPos = p.yyPos;
		xxVel = p.xxVel;
		yyVel = p.yyVel;
		mass = p.mass;
		imgFileName = p.imgFileName;
	}

	public double calcDistance (Planet s) {
		double d = Math.sqrt(Math.pow((xxPos - s.xxPos), 2) + Math.pow((yyPos - s.yyPos), 2));
		return d;
	}

	public double calcForceExertedBy (Planet s) {
		double g = 6.67E-11;
		double rr = Math.pow(calcDistance(s), 2);
		double f = g * mass * s.mass / rr;
		return f;
	}

	public double calcForceExertedByX (Planet s){
		double g = 6.67E-11;
		double r = calcDistance(s);
		double dx = s.xxPos - xxPos;
		double fx = (calcForceExertedBy(s) * dx / r);
		return fx;
	}

	public double calcForceExertedByY (Planet s){
		double g = 6.67E-11;
		double r = calcDistance(s);
		double dy = s.yyPos - yyPos;
		double fy = (calcForceExertedBy(s) * dy / r);
		return fy;
	}

	public double calcNetForceExertedByX(Planet[] allPlanets) {
		int c = 0;
		double nfx = 0;
		while (c != allPlanets.length) {
			if (this != allPlanets[c]) {
				nfx = nfx + calcForceExertedByX(allPlanets[c]);
				c = c + 1;
			} else {
				c = c + 1;
			}
		}
		return nfx;
		}
		
	public double calcNetForceExertedByY(Planet[] allPlanets) {
		int c = 0;
		double nfy = 0;
		while (c != allPlanets.length) {
			if (this != allPlanets[c]) {
				nfy = nfy + calcForceExertedByY(allPlanets[c]);
				c = c + 1;
			} else {
				c = c + 1;
			}
		}
		return nfy;
		}

	public void update (double dt, double fX, double fY) {
		double ax = fX/mass;
		double ay = fY/mass;
		double xvtemp = xxVel;
		double yvtemp = yyVel;
		double xptemp = xxPos;
		double yptemp = yyPos;
		xxVel = xvtemp + (dt * ax);
		yyVel = yvtemp + (dt * ay);
		xxPos = xptemp + (dt * xxVel);
		yyPos = yptemp + (dt * yyVel);
	}

	public void draw () {
		StdDraw.picture(xxPos, yyPos, imgFileName);
	}
}
