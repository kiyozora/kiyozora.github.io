public class NBody{


	public static void main(String[] args){
		double time = 0;
		double T = Double.parseDouble(args[0]);
		double dt = Double.parseDouble(args[1]);
		String fileName = args[2];
		double radius = readRadius(fileName);
		StdDraw.setScale(-radius, radius);
		StdDraw.picture(0, 0, "images/starfield.jpg");
		Planet[] allPlanets = readPlanets(fileName);
		for(int i = 0; i < allPlanets.length; i++){
			allPlanets[i].draw();
		}


		while(time < T){
			double[] xForces = new double[allPlanets.length];
			double[] yForces = new double[allPlanets.length];
			for(int r = 0; r < allPlanets.length; r++){
				xForces[r] = allPlanets[r].calcNetForceExertedByX(allPlanets);
				yForces[r] = allPlanets[r].calcNetForceExertedByY(allPlanets);
			}
			for(int k = 0; k < allPlanets.length; k++){
				allPlanets[k].update(dt, xForces[k], yForces[k]);
			}
			StdDraw.picture(0, 0, "images/starfield.jpg");
			for(int l = 0; l < allPlanets.length; l++){
				allPlanets[l].draw();
		}
			StdDraw.show(10);
			time += dt;

		}
	

		StdOut.printf("%d\n", allPlanets.length);
		StdOut.printf("%.2e\n", radius);
		for (int i = 0; i < allPlanets.length; i++) {
			StdOut.printf("%11.4e %11.4e %11.4e %11.4e %11.4e %12s\n",
   			allPlanets[i].xxPos, allPlanets[i].yyPos, allPlanets[i].xxVel, allPlanets[i].yyVel, allPlanets[i].mass, allPlanets[i].imgFileName);	
}	
		
	}


	public static double readRadius(String text){
		In in = new In(text);
		int size = in.readInt();
		double radius = in.readDouble();
		return radius;

	}

	public static Planet[] readPlanets(String text){
		In in = new In(text);
		int size = in.readInt();
		Planet[] planets = new Planet[size];
		double g = in.readDouble();
		for(int i = 0; i < size; i++){
			double xxPos = in.readDouble();
			double yyPos = in.readDouble();
			double xxVel =in.readDouble();
			double yyVel =in.readDouble();
			double mass =in.readDouble();
			String imgFileName = in.readString();
			Planet p = new Planet(xxPos, yyPos, xxVel, yyVel, mass, imgFileName);
			planets[i] = p;

		}
		return planets;


	}


}
